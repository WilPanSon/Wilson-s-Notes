\subsubsection{Probabilistic Tools}
We will also make use of Markov's Inequality to bound probabilities.

\begin{thm}[Markov's Inequality]
    If $X$ is a non-negative random variable and $k > 0$ then:
    \[
    \mathbb{P}[X\geq k\cdot \mathbb{E}[X]]\leq \frac{1}{k}.
    \] 
\end{thm}
\subsection{100 Prisoners Problem:} In a prison, there are $100$ prisoners and there is a warden who want to retire. He decides to play a game.
    \begin{enumerate}
        \item In a room there are $100$ boxes, In these boxes, is a permutation of the numbers $1,2,...,100$. The boxes are closed and the prisoners cannot see the numbers within the boxes.
        \item The prisoners will each enter and play a mini-game. For the $i$-th prisoner, they open $50$ boxes and wins by finding the number $i$.
        \item After a prisoner leaves, the boxes are closed again, and the state of the room is restored to exactly how it was before they entered. Prisoners cannot communicate after the game begins.
        \item Prisoners win the game (and survive) if and only if \textbf{every} prisoner finds their own number. Otherwise, they are all executed.
    \end{enumerate}
\noindent \textbf{Naive Algorithm:} Each prisoner opens $50$ random boxes then \[
\mathbb{P}[\text{Victory}] = \left(\frac{1}{2}\right)^{100} \approx 8 \times 10^{-31}.
\]
\subsection{Cycle Following Algorithm}

We propose a strategy that utilizes the permutation structure inside the boxes.\\
\textbf{Cycle Algorithm:} 
    Consider the algorithm where person $i$ opens box $i$, sees $j$, then open box $j$, see $k$ and so on. This continues until they find their number or exhaust their $50$ tries.

\begin{lem}
    Under the Cycle Algorithm, prisoner $i$ succeeds if and only if $i$ belongs to a cycle of length less than or equal to 50. Consequently, the team achieves victory if and only if the permutation contains no cycle of length strictly greater than 50.
\end{lem}
\subsubsection{A Warm-up Bound using Markov's Inequality}
Before calculating the exact probability, let us try to bound the failure probability of a slightly easier version of the game using Markov's Inequality.

\begin{lem}
    Consider a variant where each prisoner is allowed to open 75 boxes. Using the Cycle Algorithm, the probability of failure is at most $1/3$.
\end{lem}
\begin{proof}
    Let $X$ be the number of prisoners who lose. We write \[
    X = \sum_{i=1}^{100} I_i \text{ where } I_i=\begin{cases}
        1 & \text{if prisoner $i$ loses} \\
        0 & \text{otherwise} 
    \end{cases}
    .\]
    First, observe that for any prisoner $i$, the probability that their specific number lies in the 25 unopened boxes is exactly $25/100 = 1/4$. Thus, $\mathbb{E}[I_i] = 1/4$. By linearity of expectation:
    \[
    \mathbb{E}[X] = \sum_{i=1}^{100} \mathbb{E}[I_i] = 100 \cdot \frac{1}{4} = 25.
    \]\\
    Now, observe the Cycle Algorithm: If prisoner $i$ loses, they must be part of a cycle of length greater than 75. As cycles are disjoint, anyone else with a number in that same cycle will also lose. Therefore, if anyone loses like when $X>0$, then strictly more than 75 people must lose so $X>75$.\\
    Using Markov's Inequality with $k=3$:
    \[
    \mathbb{P}[\text{Non-victory}] = \mathbb{P}[X > 0] = \mathbb{P}[X \geq 76] \leq \mathbb{P}[X \geq 75] \leq \frac{\mathbb{E}[X]}{75} = \frac{25}{75} = \frac{1}{3}.
    \]
\end{proof}
\subsubsection{Exact Probability Calculation}

\begin{thm}
    The Cycle Algorithm achieves a victory probability of $\mathbb{P}[\text{Victory}] \approx 0.31$.
\end{thm}
\begin{proof}
    The prisoners fail if and only if there exists a cycle of length $\ell > 50$. Since the sum of cycle lengths is 100, there can be at most one cycle of length greater than 50. Thus, the events of having a cycle of length $\ell$ (for $\ell > 50$) are mutually exclusive.
    \[
    \mathbb{P}[\text{Non-victory}] = \mathbb{P}[\exists \text{ cycle of length } > 50] = \sum_{\ell=51}^{100} \mathbb{P}[\exists \text{ cycle of length } \ell].
    \]
    To calculate the number of permutations of $N=100$ elements that contain a cycle of length exactly $\ell$.
    \begin{enumerate}
        \item Choose $\ell$ elements to be in the cycle: $\binom{N}{\ell}$ ways
        \item Arrange these $\ell$ elements into a cycle: $(\ell-1)!$ ways
        \item Permute the remaining $N-\ell$ elements arbitrarily: $(N-\ell)!$ ways
    \end{enumerate}
    The total number of such permutations is:
    \[
    \binom{N}{\ell} \cdot (\ell - 1)! \cdot (N - \ell)! = \frac{N!}{\ell!(N-\ell)!} (\ell - 1)! (N - \ell)! = \frac{N!}{\ell}.
    \]
    The probability that a random permutation contains a cycle of length $\ell$ is therefore:
    \[
     \frac{N! / \ell}{N!} = \frac{1}{\ell}.
    \]
    Summing these probabilities:
    \[
     \mathbb{P}[\text{Non-victory}] = \sum_{\ell=51}^{100} \frac{1}{\ell} = H_{100} - H_{50} \approx \ln(100) - \ln(50) = \ln(2) \approx 0.69.
    \]
    Thus, the probability of victory is $1 - 0.69 \approx 0.31$.
\end{proof}
\begin{thm}
    The previous algorithm is optimal. 
\end{thm}
\begin{proof}
    Consider a second version of the game:
    \begin{enumerate}
        \item Prisoner 1 enters, opens 50 boxes 
        \item The difference between the games is that the boxes \textbf{remain open} for subsequent prisoners
        \item Prisoner $i$ enters. They win immediately if their box was already opened. Otherwise they may open more boxes until 50 total boxes (including those opened by previous prisoners) have been opened in the room
    \end{enumerate}
    We rely on the following two lemmas to complete the theorem.
\end{proof}
\begin{lem}
    The Cycle Algorithm achieves victory in Game 2 if and only if there are no cycles of length greater than 50.
\end{lem}
\begin{proof}
    Consider when there are no cycles of length $>50$, the Cycle Algorithm wins Game 1, and since Game 2 provides more information as boxes stay opened, it also wins Game 2.
    Conversely, suppose there is a cycle of length $>50$. The first prisoner in that cycle to enter the room will be forced to open the boxes in that cycle, since the boxes were closed before they entered or if open, did not contain their number, they must follow the cycle. They will run out of their 50 openings before finding their number. Thus, the team fails. Therefore, the performance of the Cycle Algorithm is identical in Game 1 and Game 2.
\end{proof}
\begin{lem}
    In Game 2, all strategies yield the same probability of victory.
\end{lem}
\begin{proof}
    In Game 2, the boxes remain open, this means the decision of which box to open next depends on the set of boxes currently revealed. However, since the permutation inside the boxes is chosen uniformly, the values hidden in the unopened boxes are a random permutation of the remaining numbers.\\ 
    Consequently, the probability that the \emph{next} box opened contains a specific required number depends only on the number of boxes remaining, not on which specific box is chosen. Regardless of the strategy used, the probability that the union of opened boxes contains the set of all prisoners' numbers is not changed.
\end{proof}
\noindent \textbf{Conclusion of Optimality:}
We have established the following chain of inequalities:
\[
\mathbb{P}(\text{Cycle wins Game 1}) \leq \mathbb{P}(\text{Optimal Strategy wins Game 1}) \leq \mathbb{P}(\text{Optimal Strategy wins Game 2}).
\]
The second inequality is true as any strategy for Game 1 is a strategy for Game 2 if we ignore the extra information, so Game 2 is strictly easier.
However, by Lemma 4.3, all strategies in Game 2 are equivalent.
\[
\mathbb{P}(\text{Optimal Strategy wins Game 2}) = \mathbb{P}(\text{Cycle wins Game 2}).
\]
And by Lemma 4.2:
\[
\mathbb{P}(\text{Cycle wins Game 2}) = \mathbb{P}(\text{Cycle wins Game 1}).
\]
Combining these we get:
\[
\mathbb{P}(\text{Cycle wins Game 1}) \leq \mathbb{P}(\text{Optimal Strategy wins Game 1}) \leq \mathbb{P}(\text{Cycle wins Game 1}).
\]
Therefore, the inequalities must be equalities, and the Cycle Algorithm is optimal for Game 1.

\subsection{Mcdiarmid's Theorem}
\begin{thm}
    Mcdiarmid's Theorem (1993) Given 3-colorable graph, we can efficiently compute a triangle-free partition. \\
    The algorithm is in each round \begin{enumerate}
        \item Find some triangle in some part
        \item Pick random vertex in the triangle
        \item Move vertex to other side
    \end{enumerate}
\end{thm}
\begin{proof}
    Pick a specific $3$ coloring of the graph then let $X=$ \# color $1$ vertices on left and $Y=$ \# color $2$ vertices on right side. Observe that the value $X+Y$ changes by $ \{-1,0,1\}$ with equal probability. Then observe a random walk in $[0,n]$ \\
    \begin{lem}
        Let $T(n)=\E[ \text{time to get to }\pm n]$ then $T(n)=n^2$, but we restrict it to powers of $2$.\\
        We can easily see the following recurrence holds and it implies the lemma \[
        T(n)=T \left( \frac{n}{2} \right)+ T \left( \frac{n}{2} \right)+\frac{1}{2}T(n)
        .\] 
    \end{lem}
    So the expected number of rounds is $O \left( n^2 \right)$
\end{proof}
\subsection{Some theorem}
\begin{thm}
    Given $G=(V,E)$ we want for each pair $x,y\in V$ to calculate $d(x,y)\pm 2$. We will do it in $O(n^{2.5}\ln n)$.
\end{thm}
\begin{proof}
    Consider the shortest path $x\to y$ 
    \begin{enumerate}
        \item[Option 1:] Path has no vertices with degree $>\sqrt{n}$. We can run BFS from each node, but exclude high degree nodes from search. So $O(|V|\cdot n^{1.5})=O(n^{2.5})$
        \item[Option 2:] There exist a high degree vertex on $x\to y$ path. Define $S=100\sqrt{n}\log n$ vertices, each of which is an independent uniform random vertex. 
    \end{enumerate}
    Claim: With good probability, every high-degree vertex is incident to at least one vertex in $S$. \\
    Observe that there exist $s\in S$ such that \[
    d(x,s)+d(s,y)\leq d(x,y)+2
    .\]
    For each $s\in S$ do BFS and this runs at $O(n^{2.5} \log n)$
    Then for each $x,y$ compute \[
    \min_{s\in S}d(x,s)+d(s,y)
    .\] 
    This also is $O(n^{2.5}\log n).$
    \begin{align*}
        \Pr[ \text{given high degree vertex } w\not \sim S]&\leq \Pr[ \text{every sample misses neigh}(c_v)]\\
        &\leq \left(  1-\frac{1}{\sqrt{n}} \right)^{100\sqrt{n}\log n}\\
        &\approx \left( \frac{1}{e} \right)^{100\log n}\leq \frac{1}{n^{100}}
    \end{align*} 
\end{proof}

\begin{lem}
    (Poor Man's Chernoff Bound) \[
    \Pr[X \geq 2k \sqrt{n}] \leq \frac{1}{2^k}
    .\] 
\end{lem}
\begin{proof}
    Note: \[
    \Pr[X\geq 2\sqrt{n}] \leq \frac{1}{4} \tag{Chebyshev's Inequality}\implies \Pr[ \text{any prefix of $X$ is } \geq 2\sqrt{n}]\leq \frac{1}{2}
    .\] 
    Let $t_i$ be the event that the prefix of $X$ reaches $2i\sqrt{n}$ then \[
    \Pr[t_i \text{ exists} | {t_{i-1} \text{ exist}}  ]\leq \frac{1}{2}
    .\] 
    So \[
    \Pr[X \geq 2k\sqrt{n}] \leq \Pr[t_k \text{ exists}] \leq \frac{1}{2^k}
    .\] 
\end{proof}
\begin{lem}
    Let $X_1,...,X_n$ be i.i.d random variables where each $X_i$ satisfies \[
    \Pr[X_i \geq j] \leq p^j
    .\] 
    Then $X=\sum_{i=1}^n X_i$ satisfies \[
    \Pr[X\geq 2n]\leq (4p)^n
    .\] 
\end{lem}
\begin{proof}
    \[
    \sum X_i \geq 2n \implies \sum \floor{X_i}\geq n \implies \exists \vec{Y}=(Y_1,...,Y_n) \text{ such that } X_i\geq Y_i \text{ for all $i$ and } \sum Y_i= n 
    .\] 
    FOr a given $\vec{Y}=(Y_1,...,Y_n)$ we have \[
    \Pr[\vec{Y} \text{ occurs}] = \Pr[X_i\geq Y_i \text{ }\forall i]= \prod_i \Pr[X_i \geq Y_i] \leq \prod_i p^{Y_i} = p^n
    .\] 
    We can encode $\vec{Y}$ as a binary string by stars and bars of length $2n.$ \[
    {2n \choose n} \leq 2^{2n}
    .\] 
    So number of such $\vec{Y}$ is at most $4^n$. So \[
    \Pr[X \geq 2n] \leq \sum_{\vec{Y}} \Pr[\vec{Y} \text{ occurs}] \leq 4^n \cdot p^n = (4p)^n
    .\] 
\end{proof}
\begin{thm}
    The (Anti-Chernoff Bound) \[
    \Pr[X\geq \frac{k}{4}\sqrt{n}] \geq \frac{1}{4^{k^2}}
    .\] 
\end{thm}
\begin{proof}
    We'll prove using the core fact \[
    \Pr[X \geq \frac{1}{4}\sqrt{n}] \geq \frac{1}{4}
    .\]

    .\] 
    If this occurs $X=\sum C_i \geq k^2 (\frac{1}{4}\sqrt{m})=\frac{k}{4}\sqrt{n}$. 
\end{proof}

\begin{thm}
    McDiarmid's Inequality: Let $X_1,...,X_n$ be independent random variables. \\
    Suppose $f$ is $1$-lipchitz for any $X_1,...,X_n$ and $\overline{X_i}$ then \[
    |f(X_1,...,X_i,...,X_n)-f(X_1,...,\overline{X_i},...,X_n)|\leq 1
    .\] 
    Then \[
    \Pr[f-\E[f]\geq k\sqrt{n}]\leq e^{-\frac{k^2}{2}}
    .\] 
\end{thm}
\begin{proof}
    Define $Y_1=\E[f|X_1]-\E[f]$, $Y_2=\E[f|X_1,X_2]-\E[f|X_1]$ and in general $Y_i=\E[f|X_1,...,X_i]-\E[f|X_1,...,X_{i-1}]$. \\
    Then \[
    \sum_{i=1}^n Y_i=\E[f|X_1,...,X_n]-\E[f]
    .\] 
    We have each $Y_i\in [-1,1]$ and we need to prove $\E[Y_i | X_1,...,X_{i-1}]=0$. \\
    Fix $X_1,...,X_{i-1}$. Let $X_i$ still be random then by law of total expectation \[
    \underbrace{\E[\overbrace{\E[f | X_1,...,X_i]}^{P}]}_{\text{random over $X_i$}}=\underbrace{\E[f| X_1,...,X_{i-1}]}_{Q}
    .\] 
    So \[
    \E[Y_i]=\E[P-Q | X_1,...,X_{i-1}]=0
    .\] 
\end{proof}

\begin{prop}
    The Cup Game: 
    \begin{enumerate}
        \item There are $n$ cups that are initially empty 
        \item In each step, the filler distributes up to $1-\epsilon$ units of water for $\epsilon\in [0,\frac{1}{2}]$
        \item The emptier removes up to $1$ unit from single cup. 
        \item The emptier's goal is to minimize backlog (amount of water in the fullest cup)
        \item Filler is a oblivious adversary that does not adapt to behavior of emptier and precommits to what it will do on each time step 
    \end{enumerate}
\end{prop}

    The greedy algorithm for the emptier is to empty up to $1$ unit from fullest cup. This was proven in 1987 by Dietz/Slaeator where Greedy achieves backlog of $O(\log n)$ and the filler can force backlog $\Omega(\log n)$ against any deterministic algorithm. The filler achieves this by 
    \begin{enumerate}
        \item Put $\frac{1-\epsilon}{n}$ water in each cup
        \item Emptier empties from some cup
        \item Filler puts $\frac{1-\epsilon}{n-1}$ water in each remaining cup 
    \end{enumerate}
    Continuing this the backlog after $n-1$ steps is \[
    (1-\epsilon) \left( \frac{1}{n}+ \frac{1}{n-1}+ \cdots + \frac{1}{2} \right)=\theta(\log n)
    .\] 
\end{rmk}

    (Bender, Foreach-Colton, Kuszmaul) Let $\epsilon\in \theta(1)$. There is a random algorithm that at any given moment, guarantees backlog $O(\log \log n)$ with high probability. 
\end{thm}
\begin{proof}
    Smoothed Greedy Algorithm for Emptier:
    \begin{enumerate}
        \item At the start, pick random i.i.d $r_1,...,r_n\in [0,1]$ and put $r_i$ water in cup $i$. 
        \item In each step, empty $1$ unit from fullest cup but if every cup has $<1$ unit then skip our turn. 
    \end{enumerate}
    Let $a_i(t)$ be the amount of water the filler puts in cup $i$ on step $t$, $b_i(t)$ be the amount of water the emptier removes from cup $i$ on step $t$ and $c_i(t)$ be the fill of cup $i$ after step $t.$
    \begin{claim}
        For all $t,i, c_i(t) \pmod 1$ is uniformly random. 
    \end{claim}
    \begin{proof}
        \[
        c_i(t)=\underbrace{r_i}_{\text{random}}+\underbrace{\sum_{\ell=1}^t a_i(\ell)}_{ \text{fixed}}-\underbrace{\sum_{\ell=1}^t b_i(\ell)}_{ \text{integer}}
        .\] 
    \end{proof}
    \noindent Define a potential function $\phi(t)=\sum_{i=1}^n \floor{c_i(t)^2}$
    \begin{lem}
        Consider time $t_1,$ let $t_0<t_1$ be most recent time that $\phi(t_0)$ was zero. With high probability, $t_1-t_0=O(\log n)$. 
    \end{lem}
    \begin{proof}

        \Pr[ \text{cup $i$ crosses an int threshold}]=a_i(t)
        .\]

        .\] 
    \end{proof}
    \noindent Consider steps $t_0+1,...,t_1-1$, all cups are initially $<1$ and consider only cups that go above $2$. The number of such cups is $\leq O(\log n)$ as $t_1-t_0=O(\log n)$ with high probability. \\

\end{proof}

\section{Feb 23: Bourgain's Embedding Theorem}

\begin{defn}
    Given $(X_1,d_1)$ and $(X_2,d_2)$, a map $\phi:X_1\to X_2$ has distortion $\alpha \geq 1$ if there exists a scaling parameter $\beta > 0$ such that for all $x, y \in X_1$:
    \[
        \beta d_1(x,y) \leq d_2(\phi(x),\phi(y)) \leq \alpha \beta d_1(x,y)
    .\]
\end{defn}

\begin{thm}[Bourgain's Theorem]
    Let $(X,d)$ be any $n$-point metric space. Then there exists an embedding $\phi:(X,d)\to (\mathbb{R}^{O(\log^2 n)}, d_{\ell_1})$ such that the distortion is $O(\log n)$.
\end{thm}

\begin{defn}
    For $i=1,\dots,\log n$ and $j=1,\dots,c\log n$ (for a sufficiently large constant $c$), define a random subset $S_{i,j} \subseteq X$ by including each $x\in X$ independently with probability $\frac{1}{2^i}$.\\
    We define the embedding $\phi: X \to \mathbb{R}^{\log n \times c\log n}$ as:
    \[
        \phi(x) = \big( d(x,S_{1,1}), d(x,S_{1,2}), \dots, d(x,S_{\log n, c\log n}) \big)
    .\]
    where $d(x,S) = \min_{s\in S} d(x,s)$.
\end{defn}

\begin{thm}
    With high probability in $n$, for all $x,y\in X$:
    \[
        \Omega(\log n) d(x,y) \leq d_{\ell_1}(\phi(x),\phi(y)) \leq c\log^2 n \, d(x,y)
    .\]
    This yields an embedding with scaling factor $\beta = \Omega(\log n)$ and distortion $\alpha = O(\log n)$.
\end{thm}

\begin{rmk}
    Recall that the $\ell_1$ distance is the sum of absolute differences across all coordinates. Thus, we need to analyze:
    \[
        d_{\ell_1}(\phi(x),\phi(y)) = \sum_{i=1}^{\log n}\sum_{j=1}^{c\log n} \left| d(x,S_{i,j}) - d(y,S_{i,j}) \right|
    .\]
    The upper bound is immediate from the triangle inequality: $|d(x,S) - d(y,S)| \leq d(x,y)$. Summing this over all $c\log^2 n$ coordinates gives the upper bound $c\log^2 n \, d(x,y)$. The core of the proof is establishing the lower bound.
\end{rmk}

\begin{proof}
    Fix $x,y\in X$. For any $z \in X$ and $r \geq 0$, let $B(z,r) = \{w\in X \mid d(z,w) \leq r\}$ be the closed ball, and let $B^\circ(z,r) = \{w\in X \mid d(z,w) < r\}$ be the open ball.\\
    For each $i$, define $\rho_i(x) = \inf\{r \mid |B(x,r)| \geq 2^i\}$. By definition, the open ball is strictly smaller than $2^i$, so $|B^\circ(x,\rho_i(x))| < 2^i$, while the closed ball satisfies $|B(x,\rho_i(x))| \geq 2^i$.\\
    Define $r_i = \max(\rho_i(x), \rho_i(y))$. Without loss of generality, assume $r_i = \rho_i(x)$. This gives us two critical observations:
    \begin{enumerate}
        \item $|B^\circ(x,r_i)| < 2^i$ (since $r_i = \rho_i(x)$).
        \item $|B(y,r_i)| \geq 2^i$ (since $r_i \geq \rho_i(y)$). Since balls only grow with radius, it also holds that $|B(y,r_{i-1})| \geq 2^{i-1}$.
    \end{enumerate}
    Define $t$ as the largest integer such that $r_t \leq \frac{d(x,y)}{3}$. Define $r_0 = 0$, and cap the sequence by defining $r_{t+1} = \frac{d(x,y)}{3}$.

    \begin{lem}
        For any $i \in \{1,\dots,t+1\}$ and $j \in \{1,\dots,c\log n\}$, with constant probability $p > 0$ (specifically $p > \frac{1}{10}$),
        \[
            |d(x,S_{i,j}) - d(y,S_{i,j})| \geq r_i - r_{i-1} \tag{$*$}
        .\]
    \end{lem}
    \noindent\textit{Proof of Lemma:} To satisfy $(*)$, it is sufficient for $S_{i,j}$ to miss the open ball $B^\circ(x,r_i)$ and hit the closed ball $B(y,r_{i-1})$.
    Since $r_i \leq \frac{d(x,y)}{3}$ and $r_{i-1} \leq \frac{d(x,y)}{3}$, the distance between the centers of these two balls is $d(x,y) \geq r_i + r_{i-1} + \frac{1}{3}d(x,y)$. Thus, $B^\circ(x,r_i)$ and $B(y,r_{i-1})$ are completely disjoint, making hitting and missing them completely independent events.
    \begin{itemize}
        \item \textbf{Miss Event:} $\Pr[S_{i,j} \cap B^\circ(x,r_i) = \emptyset] = \left(1 - \frac{1}{2^i}\right)^{|B^\circ(x,r_i)|} \geq \left(1 - \frac{1}{2^i}\right)^{2^i} \approx \frac{1}{e}$.
        \item \textbf{Hit Event:} $\Pr[S_{i,j} \cap B(y,r_{i-1}) \neq \emptyset] = 1 - \left(1 - \frac{1}{2^i}\right)^{|B(y,r_{i-1})|} \geq 1 - \left(1 - \frac{1}{2^i}\right)^{2^{i-1}} \approx 1 - \frac{1}{\sqrt{e}}$.
    \end{itemize}
    By independence, the probability that both occur is at least $\frac{1}{e} \left(1 - \frac{1}{\sqrt{e}}\right) \approx 0.14 > \frac{1}{10}$. When this happens, $d(x, S_{i,j}) \geq r_i$ and $d(y, S_{i,j}) \leq r_{i-1}$, so the absolute difference is at least $r_i - r_{i-1}$.

    \begin{claim}
        For any $i \leq t+1$, let $Z_i$ be the number of indices $j$ such that $(*)$ holds. Then $Z_i \geq \Omega(\log n)$ with high probability.
    \end{claim}
    \noindent\textit{Proof of Claim:} $Z_i$ is the sum of $c\log n$ independent 0-1 random variables, each with mean $\geq \frac{1}{10}$. The expected value is $\mu \geq \frac{c\log n}{10}$. By the Chernoff bound, for a sufficiently large constant $c$, the probability that $Z_i < \frac{c\log n}{20}$ is exponentially small ($o(1/n^3)$). Applying a union bound over all pairs $x,y$ and all $i$, this bound holds uniformly with high probability. Let $K = \frac{c}{20}$, so $Z_i \geq K\log n$.

    \medskip\noindent
    Using the claim, we can now telescope the sum for the lower bound:
    \begin{align*}
        d_{\ell_1}(\phi(x),\phi(y)) &= \sum_{i=1}^{\log n}\sum_{j=1}^{c\log n} \left| d(x,S_{i,j}) - d(y,S_{i,j}) \right| \\
        &\geq \sum_{i=1}^{t+1} \sum_{j : (*)\text{ holds}} (r_i - r_{i-1}) \\
        &\geq \sum_{i=1}^{t+1} (K\log n) (r_i - r_{i-1}) \\
        &= K\log n \sum_{i=1}^{t+1} (r_i - r_{i-1}) \\
        &= K\log n (r_{t+1} - r_0) \\
        &= K\log n \frac{d(x,y)}{3} \\
        &= \Omega(\log n) d(x,y)
    .\end{align*}
    This completes the proof of the lower bound, showing the overall distortion is bounded by $O(\log n)$.
\end{proof}

\begin{defn}[Tree Metric]
    Let $(V,d)$ be a metric space. A tree metric is defined by a tree $T = (V_T, E_T)$ with non-negative edge weights, such that $V \subseteq V_T$. The tree distance $d_T(x,y)$ between any two nodes $x,y \in V$ is the sum of the edge weights along the unique simple path connecting them in $T$. 
    
    We say $T$ embeds $(V,d)$ with distortion $\alpha \geq 1$ if for all $x,y \in V$:
    \[
        d(x,y) \leq d_T(x,y) \leq \alpha d(x,y)
    .\]
\end{defn}
\begin{thm}
    Let $(X,d)$ be $n$ point space. Suppose $\forall x\neq y\in X$, $d(x,y)\in [1,n^2]$. We will embed $X$ into a tree metric $(T,d_T)$ with 
    \begin{enumerate}
        \item depth of tree is $O(\log n)$.
        \item For any root-leaf path, edges weights are decreasing powers of $2$.
        \item Points in $X$ are mapped to leaves of $T$
    \end{enumerate}
    There exist randomized embedding $\phi:X\to (T,d_T)$ such that $\forall x,y\in X$, $d_T(\phi(x),\phi(y))\geq d(x,y)$ and $\E[d_T(\phi(x),\phi(y))]\leq O(\log n)d(x,y)$
\end{thm}
\begin{proof}
    The algorithm is as follows:
    \begin{enumerate}
        \item Pick random permutation of elements of $X$ say $\pi=\pi_1,...,\pi_n$
        \item Pick uniformly random $\beta\in [1,2)$ and define $\beta_i:=2^{i-1}\beta$ and for each $x\in X$ and $i=0,...,2\log n$ and define $center_i(x):=$ first $\pi_j$ to satisfy $d(x,\pi_j)\leq \beta_i$
        \item Construct as follows: Each node represents sets of elements with root at level $2\log n+1=$ all of $X$. For each node $w$ at level $i+1$, if node $>1$ element, group $w$'s elements $z$ by $center_i(z)$. Edges from level $i+1$ to level $i$ has weight $2^{i}\beta$
    \end{enumerate}
    Analysis:\\
    We claim: For $x,y\in X$, $d_T(\phi(x),\phi(y))\geq d(x,y)$.
    To prove the claim, observe at level $i$, if $x,y$ in same node then $d(x,y)\leq 2\beta_i\leq 2^{i+1}$. Let $i=$ highest level path goes through between $x$ and $y$. $d_T(\phi(x),\phi(y))\geq 2\cdot 2^i\geq d(x,y)$. \\
    Property: $\E[d_T(\phi(x),\phi(y))]\leq O(\log n)d(x,y)$\\
    To prove this property, $x:=\phi(x), y:=\phi(y)$,we can trivially say $d_T(x,y)\leq O( \text{largest edge weight along path})$. This is upper bounded by $\sum_{i=0}^{2\log n} 2^{i+1} \mathbb{I}[center_i(x)\neq center_i(y)]$. Taking the expectation we get 
    \begin{align*}
        \E[d_T(x,y)]&\leq \sum_{i=0}^{2\log n} \int_{r=2^{i-1}}^{2^i} \Pr[B_i=r]\Pr[center_i(x)\neq center_i(y)|B_i=r]\cdot 2^{i-1}dr\\
        &=\sum_{i=0}^{2\log n} \int_{r=2^{i-1}}^{2^i} \frac{1}{2^{i-1}}\Pr[center_i(x)\neq center_i(y)|B_i=r]\cdot 2^{i-1}dr\\
        &=\sum_{i=0}^{2\log n} \Pr[center_i(x)\neq center_i(y)|B_i=r]dr \tag{$*$}
     \end{align*}
    Say that $w\in X$ `cuts' $(x,y)$ at level $i$ if we look through $\pi_1,...$, the first of $ \{center_i(x),center_i(y)\}$ to appear is $w$ implies $w$ cuts $(x,y)$ and the elements are non-equal. \\
    Let $w_1,w_2,...,w_n$ be elements of $X$ sorted by distance from $w_i$ to $ \{x,y\}$. 
    \[
    *\leq \sum_{j=1}^{n}\sum_{i=0}^{2\log n}\int_{r=2^{i-1}}^{2^i}\Pr[w_j\text{ cuts } (x,y) \text{ at level } i|B_i=r]dr
    .\] 
    For $w_j$ to cut $(x,y)$ suppose wlog $d(v_j,x)\leq d(v_j,y)$ we need 
    \begin{enumerate}
        \item $d(w_j,x)\leq \beta_i\leq d(w_j,y)$ 
        \item $w_j$ has to appear before $w_1,...,w_{j-1}$ and this occurs with probability $\frac{1}{j}$
    \end{enumerate}
    So we can bound the probability as \[
    \leq \sum_{j=1}^n \int_{d(w_j,x)}^{d(w_j,y)} \frac{1}{j}dz=\sum_{j=1}^n \frac{1}{j}\left( d(w_j,y)-d(w_j,x) \right)\leq d(x,y) O(\log n) 
    .\] 
\end{proof}

\textbf{Question:} Given arbitrary wheel configuration with $n$ wheels in both front and rear quarter. can we always build a track such that the train can go along it and don't use much track. \\\\
We'll denote the wheels by the set $W\subseteq \{0,1,...,\ell\}$ with $|W|=n$. For the track, $T\subseteq \{1,2,...,2\ell \}$ and track $T$ is valid if $\forall i\in \{1,2,...,\ell\}$, $(W+i)\cap T\neq \emptyset$. 
\begin{thm}
    If $W$ is an arithmetic progression, there exist valid $T$ of size $O \left( \frac{\ell}{n} \right)$
\end{thm}
\begin{thm}
    For any $W, \exists$ valid $T$ such that $|T|= O \left( \frac{\ell \log n}{n} \right)$. 
\end{thm}
\begin{thm}
    $\exists W$ such that $\forall$ valid $T$ such that $|T|=\Omega \left( \frac{\ell \log n}{n} \right)$.
\end{thm}
\begin{proof}We'll use probabilistic method to show such a $T$ exists. 
    \textbf{Step 1:} Let $T_0$ be the track where we include each pillar, $i\in [2\ell]$ with probability $\frac{\ln n}{n}$. \\
    \textbf{Step 2:} Define $F= \{i\in [2\ell] | (W+i)\cap T_0=\emptyset\}$. \\
    Suppose for simplicity that $0\in W$, define $T=T_0\cup F$, observe that $T$ is valid. Note that since $0\in W$, $T_0\cap F=\emptyset$. 
    \\
    By linearity, 
    \[
    \E[|T|]=\underbrace{\E[|T_0|]}_{2\ell \frac{\ln n}{n}}+\E[|F|]
    .\] 
    To compute $\E[|F|]$, 
    \[
    \E[|F|]=\sum_{i=1}^{2\ell} \Pr[(W+i)\cap T_0=\emptyset]=\sum_{i=1}^{2\ell}\prod_{j\in W}\Pr[j+i\notin T_0]=\sum_{i=1}^{2\ell} \left( 1- \frac{\ln n}{n} \right)^{n}=\sum_{i=1}^{2\ell} \left( \frac{1}{e} \right)^{\ln n}=\frac{2\ell}{n}
    .\] 
    Thus we're done. 
\end{proof}
\begin{lem}
    Fix $T$ of size $\frac{\ln n}{10}$, construct \[
        W = \left \{\text{include each possible wheel in $[0,2n]$ independently with probability $\frac{1}{2}$}\right \}
    .\] 
    Then $\Pr[T \text{ valid for } W]\leq e^{-\Omega(\sqrt{n})}$
\end{lem}
\begin{proof}
    Define $F= \{i\in [\ell]|(W+i)\cap T=\emptyset\}$ then we can say $\Pr[T \text{ valid}]=\Pr[|F|=0]$.
    \[
    \E[|F|]=\sum_{i=1}^\ell \Pr[(F-i)\cap W=\emptyset]\geq \sum_{i=1}^\ell \left( \frac{1}{2} \right)^{|T|}=\sum_{i=1}^\ell \frac{1}{2^{\ln n/10}}=2
    .\] 
\end{proof}
\begin{proof}
    To show this lemma implies the theorem:
    \[
    \Pr[ \text{any $T$ of size $\frac{\log n}{10}$ is valid}]\leq \sum_{T \text{of size} \frac{\ln n}{10}} e^{-\Omega(\sqrt{n})}\leq (4n)^{\frac{\ln n}{10}} e^{-\Omega(\sqrt{n})}=e^{-\Omega(\sqrt{n})}\leq \frac{1}{4}
    .\] 
    We have $\Pr[|W|<n]\leq \frac{1}{2}$ so 
    \[
    \Pr[ \text{either $|W|<n$ or $\exists$ valid $T$}]\leq \frac{3}{4}
    .\] 
    This implies $\exists W$ of size at least $n$ such that no valid $T$ of size $\frac{\ln n}{10}$
\end{proof}
