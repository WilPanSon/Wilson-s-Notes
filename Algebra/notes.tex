\section{Lecture 1: Group Actions \& The Orbit-Stabilizer Theorem (Jan 12)}

\subsection{Permutations and Automorphisms}

\begin{defn}[Symmetric Group]
    Let $X$ be a set. The set of all permutations (bijections) of $X$ is denoted by $\Sym_X$ (or sometimes $\Sigma_X$). Under function composition, $\Sym_X$ forms a group.
\end{defn}

\begin{defn}[Automorphism Group]
    Let $(G, \cdot)$ be a group. An automorphism of $G$ is a bijection $\phi: G \to G$ that is also a homomorphism, i.e.,
    \[ \phi(g_1 g_2) = \phi(g_1)\phi(g_2) \quad \forall g_1, g_2 \in G. \]
    The set of all such automorphisms is denoted by $\Aut(G)$. It forms a group under composition.
\end{defn}

\subsection{Group Actions}

\begin{defn}[Group Action]
    Let $G$ be a group and $X$ be a set. An \textbf{action} of $G$ on $X$ is a homomorphism $\phi: G \to \Sym_X$.
    
    We typically write the action as $g \cdot x \coloneqq \phi(g)(x)$. This notation satisfies two axioms (equivalent to the homomorphism property):
    \begin{enumerate}
        \item \textbf{Identity:} $1 \cdot x = x$ for all $x \in X$.
        \item \textbf{Compatibility:} $g_1 \cdot (g_2 \cdot x) = (g_1 g_2) \cdot x$ for all $g_1, g_2 \in G, x \in X$.
    \end{enumerate}
\end{defn}

\begin{thm}[Equivalence Relation on $X$]
    Let $G$ act on $X$. Define a relation $\sim$ on $X$ by:
    \[ x_1 \sim x_2 \iff \exists g \in G \text{ such that } g \cdot x_1 = x_2. \]
    Then $\sim$ is an equivalence relation.
\end{thm}
\begin{proof}
    We check the properties:
    \begin{enumerate}
        \item \textbf{Reflexive:} $1 \cdot x = x \implies x \sim x$.
        \item \textbf{Symmetric:} If $g \cdot x_1 = x_2$, acting by $g^{-1}$ gives $g^{-1} \cdot (g \cdot x_1) = g^{-1} \cdot x_2 \implies (g^{-1}g) \cdot x_1 = g^{-1} \cdot x_2 \implies x_1 = g^{-1} \cdot x_2$. Thus $x_2 \sim x_1$.
        \item \textbf{Transitive:} If $g \cdot x_1 = x_2$ and $h \cdot x_2 = x_3$, then $h \cdot (g \cdot x_1) = x_3 \implies (hg) \cdot x_1 = x_3$. Thus $x_1 \sim x_3$.
    \end{enumerate}
\end{proof}

\subsection{Orbits and Stabilizers}

The equivalence classes under the relation $\sim$ partition the set $X$. These classes are called \textbf{orbits}.

\begin{defn}[Orbit]
    For $x \in X$, the orbit of $x$ is the set of all places $x$ can be moved to by $G$:
    \[ \mathcal{O}_x = \text{Orb}(x) = \{g \cdot x : g \in G\}. \]
\end{defn}

\begin{defn}[Stabilizer]
    For $x \in X$, the stabilizer of $x$ is the set of elements in $G$ that fix $x$:
    \[ G_x = \Stab(x) = \{g \in G : g \cdot x = x\}. \]
    Note that $G_x$ is a subgroup of $G$ (denoted $G_x \leq G$).
\end{defn}

\begin{thm}[\textbf{Orbit-Stabilizer Theorem}]
    Let $G$ act on $X$. For any $x \in X$, there is a bijection between the orbit $\mathcal{O}_x$ and the set of left cosets $G/G_x$. Consequently:
    \[ |\mathcal{O}_x| = [G : G_x]. \]
    If $G$ is finite, $|G| = |\mathcal{O}_x| \cdot |G_x|$.
\end{thm}
\begin{proof}
    Define a map $\psi: G/G_x \to \mathcal{O}_x$ by $\psi(gG_x) = g \cdot x$.
    \begin{enumerate}
        \item \textbf{Well-defined:} Suppose $g_1 G_x = g_2 G_x$. Then $g_1^{-1}g_2 \in G_x$, so $(g_1^{-1}g_2) \cdot x = x$, implying $g_2 \cdot x = g_1 \cdot x$.
        \item \textbf{Injectivity:} If $\psi(g_1 G_x) = \psi(g_2 G_x)$, then $g_1 \cdot x = g_2 \cdot x$. Multiplying by $g_1^{-1}$, we get $x = g_1^{-1}g_2 \cdot x$, so $g_1^{-1}g_2 \in G_x$, implying $g_1 G_x = g_2 G_x$.
        \item \textbf{Surjectivity:} By definition, any $y \in \mathcal{O}_x$ is of the form $g \cdot x$ for some $g$, which is exactly $\psi(gG_x)$.
    \end{enumerate}
    Thus, $\psi$ is a bijection.
\end{proof}

\section{Lecture 2: Cauchy's Theorem \& Conjugation (Jan 14)}

\subsection{Cauchy's Theorem}

\begin{lem}
    If $X$ is a finite set and $G$ is a $p$-group (a group of order $p^k$) acting on $X$, then:
    \[ |X| \equiv |X^{G}| \pmod{p}, \]
    where $X^G = \{x \in X : g \cdot x = x, \forall g \in G\}$ is the set of fixed points.
\end{lem}

\begin{thm}[Cauchy's Theorem]
    If $G$ is a finite group and $p$ is a prime dividing $|G|$, then $G$ has an element of order $p$.
\end{thm}
\begin{proof}[Proof (McKay's Proof)]
    Let $X$ be the set of $p$-tuples of elements of $G$ whose product is the identity:
    \[ X = \{(g_1, \dots, g_p) \in G^p : g_1 g_2 \cdots g_p = 1\}. \]
    Notice that $g_p$ is uniquely determined by the first $p-1$ elements via $g_p = (g_1 \cdots g_{p-1})^{-1}$, so $|X| = |G|^{p-1}$. Since $p \mid |G|$ and $p-1 \geq 1$, we have $p \mid |X|$. \\
    Let $\Z_p$ (cyclic group of order $p$) act on $X$ by cyclic shift:
    \[ \sigma \cdot (g_1, g_2, \dots, g_p) = (g_2, g_3, \dots, g_p, g_1). \]
    This is a valid action because if $g_1 \cdots g_p = 1$, then $g_2 \cdots g_p = g_1^{-1}$, so $(g_2 \cdots g_p) g_1 = g_1^{-1} g_1 = 1$, which shows the shifted tuple is still in $X$.\\
    By the Lemma, $|X| \equiv |X^{\Z_p}| \pmod{p}$. Fixed points are exactly tuples $(a, a, \dots, a)$ such that $a^p = 1$. Since $(1, 1, \dots, 1) \in X^{\Z_p}$, the set of fixed points is non-empty. Since $p \mid |X|$ and $|X| \equiv |X^{\Z_p}| \pmod p$, we have $p \mid |X^{\Z_p}|$. Therefore $|X^{\Z_p}| \geq p$, so there must exist some $a \neq 1$ with $a^p = 1$; this element has order exactly $p$.
\end{proof}

\subsection{Conjugation}

Conjugation is a specific action of $G$ on itself.

\begin{defn}
    For $g, h \in G$, the \textbf{conjugate} of $g$ by $h$ is $g^h \coloneqq hgh^{-1}$.
    The map $\phi_h: G \to G$ defined by $g \mapsto hgh^{-1}$ is an automorphism (an inner automorphism).
\end{defn}

\begin{defn}[Classes and Centralizers]
    Let $G$ act on itself by conjugation ($h \cdot g = hgh^{-1}$).
    \begin{itemize}
        \item The orbit of $g$ is the \textbf{Conjugacy Class} of $g$: $\text{Cl}(g) = \{hgh^{-1} : h \in G\}$.
        \item The stabilizer of $g$ is the \textbf{Centralizer} of $g$: $C_G(g) = \{h \in G : hg = gh\}$.
    \end{itemize}
    By Orbit-Stabilizer: $|\text{Cl}(g)| = [G : C_G(g)]$.
\end{defn}

\begin{defn}[Center of $G$]
    The \textbf{Center}, $Z(G)$, is the set of elements that commute with everything:
    \[ Z(G) = \{z \in G : zg = gz, \forall g \in G\} = \bigcap_{g \in G} C_G(g). \]
    $Z(G)$ is the kernel of the conjugation homomorphism $G \to \Aut(G)$. Thus $Z(G) \normal G$.
\end{defn}

\begin{defn}[Normalizer]
    Let $H \leq G$. The \textbf{Normalizer} of $H$ in $G$ is:
    \[ N_G(H) = \{g \in G : gHg^{-1} = H\}. \]
    $N_G(H)$ is the largest subgroup of $G$ in which $H$ is normal.
\end{defn}

\section{Lecture 3: Sylow Preliminaries \& Isomorphism Theorems (Jan 16)}

\subsection{Product of Subgroups}

\begin{defn}
    Let $K, N \leq G$. Define $KN = \{kn : k \in K, n \in N\}$.
    \begin{itemize}
        \item In general, $KN$ is not a subgroup.
        \item If $N \normal G$ (or just $N \subseteq N_G(K)$), then $KN$ is a subgroup.
        \item Size formula: $|KN| = \frac{|K||N|}{|K \cap N|}$.
    \end{itemize}
\end{defn}

\begin{thm}[Second Isomorphism Theorem]
    Let $K \leq G$ and $N \normal G$. Then $K \cap N \normal K$, and
    \[ \frac{KN}{N} \cong \frac{K}{K \cap N}. \]
\end{thm}
\begin{proof}
    Consider the natural projection $\pi: G \to G/N$. Restrict it to $K$, i.e., $\phi = \pi\res_K : K \to G/N$.
    The image of $\phi$ is $\{kN : k \in K\} = KN/N$.
    The kernel of $\phi$ is $\{k \in K : kN = N\} = \{k \in K : k \in N\} = K \cap N$.
    By the First Isomorphism Theorem, $K / \ker(\phi) \cong \text{Im}(\phi)$.
\end{proof}

\subsection{Sylow Definitions}

\begin{defn}[$p$-group]
    Let $p$ be a prime. A group $G$ is a $p$-group if every element has order a power of $p$. For finite groups, this is equivalent to $|G| = p^k$.
\end{defn}

\begin{defn}[Sylow $p$-subgroup]
    Let $|G| = p^n m$ where $p \nmid m$. A subgroup $P \leq G$ is called a \textbf{Sylow $p$-subgroup} if $|P| = p^n$.
    \[ \text{Syl}_p(G) = \{P \leq G : P \text{ is a Sylow } p\text{-subgroup}\}. \]
\end{defn}

\section{Lecture 4: The Sylow Theorems (Jan 21)}

\begin{thm}[\textbf{The Sylow Theorems}]
    Let $G$ be a finite group of order $p^n m$ where $p \nmid m$.
    \begin{enumerate}
        \item \textbf{Existence:} $\text{Syl}_p(G) \neq \emptyset$. (There exists a subgroup of order $p^n$).
        \item \textbf{Conjugacy:} Any two Sylow $p$-subgroups are conjugate in $G$. That is, if $P, Q \in \text{Syl}_p(G)$, $\exists g \in G$ such that $gPg^{-1} = Q$.
        \item \textbf{Number:} Let $n_p = |\text{Syl}_p(G)|$. Then:
            \begin{itemize}
                \item $n_p \equiv 1 \pmod{p}$.
                \item $n_p \mid m$ (equivalently $n_p \mid |G|$).
            \end{itemize}
    \end{enumerate}
\end{thm}
\begin{proof}
    Let $p$ be a prime such that $p \Bigm| |G|$. Define the set of all $p$-subgroups:
    \[
    \Sigma = \{ H : H \leq G, |H| = p^n \text{ for some } n > 0 \}
    .\]
    
    Define $\Omega$ as the set of maximal elements in $\Sigma$ under inclusion:
    \[
    \Omega = \{ H : H \in \Sigma, \text{ there is no } K \in \Sigma \text{ such that } H \subsetneq K \}
    .\]
    
    We let $G$ act on the set of subgroups $\{H : H \leq G\}$ by conjugation. Since conjugation is an isomorphism ($H^g \simeq H$), it preserves the order of subgroups. Thus:
    \[ H \in \Sigma \iff H^g \in \Sigma \quad \text{and} \quad H \in \Omega \iff H^g \in \Omega. \]
    Therefore, $G$ acts on $\Omega$ by $g \cdot H = H^g$.

    \subsubsection*{Claim 1: Let $H \in \Omega$. Consider the action of $H$ on $\Omega$ by conjugation. Then $H$ is the \emph{unique} fixed point of this action.}
    \textit{Proof of Claim:}
    \begin{itemize}
        \item \textbf{Existence:} It is trivial to see that $H$ is a fixed point, as $H^h = hHh^{-1} = H$ for any $h \in H$.
        \item \textbf{Uniqueness:} Let $K \in \Omega$ be a fixed point of the action of $H$.
        \[
        K \text{ is fixed by } H \iff K^h = K \quad \forall h \in H \iff H \leq N_G(K).
        \]
        Since $K \normal N_G(K)$, and $H \leq N_G(K)$, the product $HK$ is a subgroup of $N_G(K)$, and thus $HK \leq G$.
        Using our counting lemma:
        \[
        |HK| = \frac{|H||K|}{|H \cap K|}.
        \]
        Since $H, K \in \Omega$, their orders are powers of $p$. Thus $|HK|$ is a power of $p$, meaning $HK \in \Sigma$.
        
        We have $H \leq HK$. Since $H$ is maximal in $\Sigma$ (by definition of $\Omega$) and $HK \in \Sigma$, it must be that $H = HK$. Similarly, $K \leq HK$ implies $K = HK$.
        Therefore, $H = K$.
    \end{itemize}

    \subsubsection*{Claim 2: Size of $\Omega$ Modulo $p$}
    \textbf{Statement:} $|\Omega| \equiv 1 \pmod{p}$.
    
    \textit{Proof of Claim:}
    Let $H \in \Omega$. We decompose $\Omega$ into disjoint orbits under the action of $H$.
    \[
    \Omega = \{H\} \cup \bigcup_{K \neq H} \Orb_H(K).
    \]
    By Claim 1, $H$ is the only fixed point (an orbit of size 1). For any $K \in \Omega$ with $K \neq H$, the stabilizer of $K$ in $H$ is a proper subgroup, so by the Orbit-Stabilizer theorem for $p$-groups, $|\Orb_H(K)|$ is divisible by $p$.
    \[
    |\Omega| = 1 + \sum (\text{multiples of } p) \implies |\Omega| \equiv 1 \pmod{p}.
    \]

    \subsubsection*{Claim 3: Transitivity (Conjugacy of Elements in $\Omega$)}
    \textbf{Statement:} Any two elements of $\Omega$ are conjugate.\\
    \textit{Proof of Claim:}
    Suppose for contradiction that $H, K \in \Omega$ are not conjugates. Let $\theta_1 = \Orb_G(H)$ and $\theta_2 = \Orb_G(K)$. Since they are distinct orbits, $\theta_1 \cap \theta_2 = \emptyset$.
    
    Consider the action of $H$ on these sets:
    \begin{itemize}
        \item On $\theta_1$: $H \in \theta_1$. By Claim 1, $H$ is the unique fixed point of $H$ in $\Omega$. Since $\theta_1 \subseteq \Omega$, $H$ is the unique fixed point in $\theta_1$.
        Thus, $|\theta_1| \equiv 1 \pmod{p}$.
        
        \item On $\theta_2$: Since $H \in \theta_1$ and $\theta_1 \cap \theta_2 = \emptyset$, we have $H \notin \theta_2$.
        Therefore, $\theta_2$ contains \emph{no} fixed points under the action of $H$ (because the only fixed point in all of $\Omega$ is $H$).
        Thus, every orbit of $H$ inside $\theta_2$ has size divisible by $p$.
        This implies $|\theta_2| \equiv 0 \pmod{p}$.
    \end{itemize}
    
    By symmetry, we can swap the roles of $H$ and $K$. Running the same argument with $K$ acting on the sets implies $|\theta_2| \equiv 1 \pmod{p}$ and $|\theta_1| \equiv 0 \pmod{p}$.
    
    This results in a contradiction (e.g., $|\theta_1| \equiv 1$ and $|\theta_1| \equiv 0$). Thus, $\theta_1 = \theta_2$, so $H$ and $K$ are conjugate.
    
    \textit{Corollary:} Since there is only one orbit, $\Omega = \theta_1$. Thus $|\Omega| = |\theta_1| \equiv 1 \pmod{p}$.

    \subsubsection*{Claim 4: Divisibility}
    \textbf{Statement:} $|\Omega| \Bigm| |G|$.\\
    \textit{Proof of Claim:}
    Let $H \in \Omega$. Since $\Omega$ is exactly the orbit of $H$ under conjugation (from Claim 3), the size of the orbit is the index of the stabilizer:
    \[
    |\Omega| = [G : \Stab_G(H)] = [G : N_G(H)] = \frac{|G|}{|N_G(H)|}.
    \]
    Thus $|\Omega|$ divides $|G|$.

    \subsubsection*{Claim 5: Identification with Sylow Subgroups}
    \textbf{Statement:} $\Omega = Syl_p(G)$.\\
    \textit{Proof of Claim:}
    We know $Syl_p(G) \subseteq \Omega$ because Sylow subgroups are maximal $p$-subgroups by definition. We must show the reverse: every $H \in \Omega$ is a Sylow $p$-subgroup.\\ 
    Let $H \in \Omega$. Assume for contradiction that $H \notin Syl_p(G)$.
    Let $|G| = p^s m$ where $p \nmid m$. Since $H$ is a $p$-group but not Sylow, $|H| = p^t$ with $t < s$. \\
    Consider the normalizer $N_G(H)$.
    From Claim 2, we know $|\Omega| = [G : N_G(H)] \equiv 1 \pmod{p}$.
    Therefore, $p$ does \emph{not} divide $[G : N_G(H)]$.
    Since $|G| = [G : N_G(H)] \cdot |N_G(H)|$, all factors of $p$ in $|G|$ must reside in $|N_G(H)|$.
    Thus, $|N_G(H)|$ is divisible by $p^s$.\\
    Consequently, the index $[N_G(H) : H] = \frac{|N_G(H)|}{|H|} = \frac{(\text{multiple of } p^s)}{p^t}$ is divisible by $p$ (since $s > t$).\\
    By Cauchy's Theorem applied to the quotient group $N_G(H)/H$, there exists a subgroup of order $p$, say $\bar{K} \leq N_G(H)/H$.
    By the Correspondence Theorem, there exists a subgroup $K$ such that $H \leq K \leq N_G(H)$ with $|K| = p \cdot |H| = p^{t+1}$.\\
    Since $K$ is a $p$-group, $K \in \Sigma$. However, $H \subsetneq K$, which contradicts the definition of $H$ as a maximal element in $\Omega$.\\
    Therefore, $H$ must already be of order $p^s$. Thus $\Omega = Syl_p(G)$.
\end{proof}

\section{Lecture 5: Normal Series \& Solvability (Jan 23)}

\subsection{Normal and Subnormal Series}

\begin{defn}
    A \textbf{subnormal series} of a group $G$ is a chain of subgroups:
    \[ \{1\} = H_0 \leq H_1 \leq \dots \leq H_k = G \]
    such that $H_i \normal H_{i+1}$ for all $i$.
    The quotient groups $H_{i+1}/H_i$ are called the \textbf{factors} of the series.
\end{defn}

\begin{defn}[Composition Series]
    A subnormal series is a \textbf{composition series} if all factors $H_{i+1}/H_i$ are \textbf{simple} groups (non-trivial groups with no normal subgroups other than $\{1\}$ and themselves).
\end{defn}

\begin{thm}[Jordan-Hölder]
    Any two composition series of a finite group $G$ are equivalent. That is, they have the same length, and their factors are isomorphic (up to reordering).
\end{thm}

\subsection{Solvable Groups}

\begin{defn}
    A group $G$ is \textbf{solvable} if it has a subnormal series where all factors $H_{i+1}/H_i$ are \textbf{abelian}.
\end{defn}

\begin{defn}[Commutator]
    The commutator of $g, h$ is $[g,h] = ghg^{-1}h^{-1}$.
    The \textbf{Derived Subgroup} (or Commutator Subgroup) $G'$ or $[G,G]$ is the subgroup generated by all commutators.
\end{defn}

\begin{prop}
    $G/N$ is abelian if and only if $[G,G] \leq N$.
    Thus, $[G,G]$ is the smallest normal subgroup such that the quotient is abelian.
\end{prop}

\begin{rmk}[Derived Series]
    Define $G^{(0)} = G$, $G^{(1)} = [G,G]$, and $G^{(i+1)} = [G^{(i)}, G^{(i)}]$.
    $G$ is solvable if and only if the derived series terminates at $\{1\}$ (i.e., $G^{(n)} = \{1\}$ for some $n$).
\end{rmk}

\section{Lecture 6: Solvable and Derived Series (Jan 26)}
\begin{lem}
    Let $\alpha\in Aut(G)$ then $\alpha([g,h])=[\alpha(g),\alpha(h)]$.
\end{lem}
\begin{lem}
    For $N \normal G$ then $G/N$ is abelian iff $[G,G]\leq N$.
\end{lem}
\begin{thm}
    If $N\normal G$ and $g,h\in G$, then $[gN,hN]=[g,h]N$. Additionally, $gN$ and $hN$ commute in $G/N$ iff $[g,h]N=N$ iff $[g,h]\in N$. 
\end{thm}
\begin{defn}
    $G$ is solvable iff there is a subnormal series $\gen{H_i : 0\leq i \leq t}$ (i.e.\ $\{1\}=H_0\leq H_1\leq\cdots\leq H_t=G$ with $H_i\normal H_{i+1}$) such that $[H_{i+1},H_{i+1}]\leq H_i$ for all $0\leq i \leq t-1$. Equivalently, each factor $H_{i+1}/H_i$ is abelian.
\end{defn}
\begin{thm}
    The following are true
    \begin{enumerate}
        \item If $G$ solvable and $K\leq G$ then $K$ is solvable
        \item If $G$ is solvable and $N\normal G$ then $G/N$ is solvable. 
        \item For $N \normal G$, G solvable iff both N and $G/N$ are solvable
    \end{enumerate}
\end{thm}
\begin{proof}
    We prove the previous theorems
    \begin{enumerate}
        \item Let $\gen{H_i : 0\leq i \leq t}$ be a series in $G$ such that $[H_{i+1},H_{i+1}]\leq H_i$ for all $0\leq i \leq t-1$. \\
        We can let $H'_i=K\cap H_i$ for $0\leq i \leq t$ and we need to verify $\gen{H'_i : 0\leq i \leq t}$ is a solvable series in $K$.
        \item Let $\gen{H_i : 0\leq i \leq t}$ be a series in $G$ such that $[H_{i+1},H_{i+1}]\leq H_i$ for all $0\leq i \leq t-1$. \\
        Let $H_i'=\phi_N[H_i]=H_iN/N$. Verify it is solvable. 
        \item The forward direction follows from (1) and (2). For the reverse direction, assume $N$ solvable and $G/N$ solvable. Then there is a subnormal series $\gen{N_i : 0\leq i \leq s}$ in $N$ with each $N_{i+1}/N_i$ abelian, and a subnormal series $\gen{\overline{H_j} : 0\leq j \leq t}$ in $G/N$ with each $\overline{H_{j+1}}/\overline{H_j}$ abelian. By the correspondence theorem, each $\overline{H_j}$ corresponds to a subgroup $H_j$ of $G$ with $N\leq H_j$, and $H_j\normal H_{j+1}$ with $H_{j+1}/H_j\cong \overline{H_{j+1}}/\overline{H_j}$ abelian. Concatenating gives
        \[
        \{1\}=N_0\normal N_1 \normal \cdots \normal N_s=N=H_0\normal H_1 \normal \cdots \normal H_t=G
        \]
        as a subnormal series with abelian factors.
    \end{enumerate}
\end{proof}
\begin{defn}
    Given $G$, the \textbf{derived series} of $G$ is given by $G^{(0)}=G$ and $G^{(i+1)}=[G^{(i)},G^{(i)}]$ for $i\geq 0$.
\end{defn}
\begin{thm}
    $G$ is solvable iff there exists $n\geq 0$ such that $G^{(n)}=\{1\}$.
    \begin{proof}
        For the backward direction, note that by construction $G^{(i)}/G^{(i+1)}$ is abelian. Setting $H_j=G^{(n-j)}$ yields a subnormal series $\{1\}=H_0\leq H_1\leq\cdots\leq H_n=G$ with abelian factors, so $G$ is solvable.
        
        For the forward direction, let $\gen{H_j : 0\leq j \leq n}$ be a subnormal series in $G$ with $H_{j+1}/H_j$ abelian. We show by induction on $i$ that $G^{(i)}\leq H_{n-i}$: for $i=0$, $G^{(0)}=G=H_n$. Assuming $G^{(i)}\leq H_{n-i}$, since $H_{n-i}/H_{n-i-1}$ is abelian, $[H_{n-i},H_{n-i}]\leq H_{n-i-1}$, so $G^{(i+1)}=[G^{(i)},G^{(i)}]\leq [H_{n-i},H_{n-i}]\leq H_{n-i-1}$. Hence $G^{(n)}\leq H_0=\{1\}$.
    \end{proof}
\end{thm}

    Suppose $G$ abelian and simple, let $a\in G$ and $a\neq 1$ then $G=\gen{a}$ and $|a|$ is prime. 
\end{thm}

\end{thm}

\begin{thm}
    For any set $X,$ there are a group $F$ and an injective function $i:X\to F$ such that \begin{enumerate}
        \item For any group $G$ and any function $f:X\to G$ there is a unique homomorphism $\phi:F\to G$ such that $\phi(i(x))=f(x)$ for all $x\in X$.
        \begin{center}
            \begin{tikzcd}
                X \arrow[r, "i", hook] \arrow[rd, "f"'] & F \arrow[d, "\phi", dashed] \\
                & G
            \end{tikzcd}
        \end{center}
        The diagram commutes: $\phi \circ i = f$. The homomorphism $\phi$ is unique.
        \item $F$ is generated by $i[X]$
    \end{enumerate}
\end{thm}
\begin{proof}
    A \textbf{word} (on alphabet $X$) is a (possibly empty) finite sequence of pairs from $X \times \Z$. We write $x_1^{n_1}x_2^{n_2}\cdots x_k^{n_k}$ for the word $\langle(x_1,n_1),(x_2,n_2),\ldots,(x_k,n_k)\rangle$. 
    \begin{eg}
        Let $X= \{x,y,z\}$. Then $w=x^0 x^{15}y^{-23}z^{27}$ is a word.
    \end{eg}
    \noindent Let $W$ denote the set of all words. Given a word, we can possibly perform a \textbf{reduction step}:
    \begin{enumerate}
        \item If the word contains an entry $(x, 0)$, remove it.
        \item If the word contains successive entries $(x, n)$ and $(x, m)$ (same $x$), replace them with $(x, n+m)$. 
    \end{enumerate}
    \begin{eg}
        $y^{15}x^3x^{-2}x^{-1}y^2z^5\to y^{15}x^1x^{-1}y^2z^5\to y^{15}x^0y^2z^5\to y^{15}y^2z^5\to y^{17}z^5$.
    \end{eg}
    \noindent Let $R$ be the set of all \textbf{reduced} words (those on which no further reductions can be performed). \\
    For any word $w,$ there is a unique reduced word $w'$ such that $w$ can be transformed into $w'$ by a finite sequence of reductions. \\
    We will not be able to show this confluence property purely combinatorially; instead, we produce a group $F$ such that every word is equivalent to a unique reduced word. \\\\
    Given $x\in X,$ define $f_{x}:R\to R$ by cases:
    \begin{enumerate}
        \item If $w$ does not begin with $x^m$ for any $m$, then $f_x(w)=xw$.
        \item If $w=x^{-1}v$, then $f_x(w)=v$.
        \item If $w=x^n v$ for $n\neq -1$, then $f_x(w)=x^{n+1}v$.
    \end{enumerate}
    Similarly define $g_x:R\to R$ by
    \begin{enumerate}
        \item If $w$ does not begin with $x^n$ for any $n$, then $g_x(w)=x^{-1}w$.
        \item If $w=x^1 v$, then $g_x(w)=v$.
        \item If $w=x^n v$ for $n\neq 1$, then $g_x(w)=x^{n-1}v$.
    \end{enumerate}
    One checks that $f_x \circ g_x = g_x \circ f_x = 1_R$. \\
    For each $x\in X$, $f_x\in \Sym_R$ (the group of permutations of $R$), and $g_x=f_x^{-1}$. Let $F$ be the subgroup of $\Sym_R$ generated by $\{f_x : x\in X\}$; its elements are products $f_{x_1}^{n_1}\cdots f_{x_t}^{n_t}$ with $x_i\in X$, $n_i\in \Z$. \\
    Let $i:X\to F$ with $i(x)=f_x$. For any word $w=x_1^{n_1}x_2^{n_2}\cdots x_t^{n_t}$, let $f_w=f_{x_1}^{n_1}f_{x_2}^{n_2}\cdots f_{x_t}^{n_t}\in F.$\\
    Key Facts:
    \begin{enumerate}
        \item If $w$ is obtained from $\overline{w}$ by a single reduction step, then $f_w=f_{\overline{w}}$.
        \item If $w$ is obtained from $\overline{w}$ by any sequence of reduction steps, then $f_w=f_{\overline{w}}$.
        \item If $w\in R$ then $f_w(\varepsilon)=w$, where $\varepsilon$ denotes the empty word.
        \item For any word $w$, $f_w(\varepsilon)$ is the unique reduced word to which $w$ can be transformed by reduction steps.
        \item If $w_1,w_2$ are reduced words and $v$ is the unique reduction of the concatenation $w_1w_2$, then $f_v=f_{w_1}f_{w_2}$.
    \end{enumerate}
    Let $F'$ be the group whose underlying set is $R$, with operation ``concatenate and reduce''. Then $F \simeq F'$ via $f_v \mapsto v$. To finish, let $G$ be a group and $h:X\to G$ a function. We want to show there is a unique homomorphism $\phi:F'\to G$ with $\phi \circ i = h$.
    \begin{enumerate}
        \item If $\phi$ exists, then
        \begin{align*}
            \phi(x_1^{n_1}x_2^{n_2}\cdots x_t^{n_t})&=\phi(i(x_1))^{n_1}\phi(i(x_2))^{n_2}\cdots \phi(i(x_t))^{n_t}\\
            &=h(x_1)^{n_1}h(x_2)^{n_2}\cdots h(x_t)^{n_t},
        \end{align*}
        which determines $\phi$ uniquely.
        \item One verifies this formula defines a homomorphism.
    \end{enumerate}  
\end{proof}
\begin{eg}
    Let $X=\{x,y\}$ and let $G$ be any group with $2$ generators, say $G=\gen{a,b}$. Let $F$ be a free group on $X$, and let $\phi: F\to G$ be the unique homomorphism such that $\phi(x)=a$ and $\phi(y)=b$. Then $\phi$ is surjective (since $G$ is generated by $a,b$), so by the First Isomorphism Theorem, $G \simeq F/\ker(\phi)$.
\end{eg}

\begin{lem}[Gauss's Lemma]

\begin{proof}
    Suppose not, and let $p$ be prime in $R$ such that $p\mid fg$ in $R[x]$. Since $p$ is prime, $P=(p)$ is a prime ideal, so $R/P$ is an integral domain. Let $\phi:R\to R/P$ be the quotient map, and define $\phi^*:R[x]\to (R/P)[x]$ by

    \]

    \]

    \[
    \prod_{i=1}^k H_i=\left(\prod_{i=1}^k a_i\right) g.
    \]

Since the $H_i$ are primitive, so is $\prod_i H_i$ (Gauss); since $g$ is primitive, comparing contents shows $A$ and $B$ are associates, so $\prod a_i\in R^\times$. WLOG $\prod a_i=1$, giving $g=\prod H_i$ as a product of irreducibles in $R[x]$. \\\\

    \[
    f=r_1\cdots r_n\, g_1\cdots g_s=r'_1\cdots r'_m\, g'_1\cdots g'_{s'}
    \]

\end{proof}

\subsection{Goal and Preliminaries}

\begin{rmk}[Note on Bilinearity]
    Let $\phi: M \times N \to P$ be a bilinear map. If $\beta: P \to Q$ is a linear map, then the composition $\beta \circ \phi$ is bilinear.
\end{rmk}

\begin{defn}
    (Universal Property of Tensor Products)\\
    \textbf{Goal:} Construct a module $M \otimes N$ and a bilinear map $\otimes: M \times N \to M \otimes N$ such that for any module $Q$ and any bilinear map $\psi: M \times N \to Q$, there exists a unique linear map $\beta: M \otimes N \to Q$ that makes the following diagram commute:
    \begin{center}
        \begin{tikzcd}
            M \times N \arrow[r, "\otimes"] \arrow[rd, "\psi"'] & M \otimes N \arrow[d, "\exists ! \beta", dashed] \\
            & Q
        \end{tikzcd}
    \end{center}
    In other words, $\psi = \beta \circ \otimes$.
\end{defn}

\begin{lem}[Easy Fact on Quotients]
    Let $A \leq B$ be submodules. Then:
    \begin{enumerate}
        \item The natural projection $\phi_A: B \to B/A$ is a homomorphism, and it is $0$ on $A$.
        \item For any module $C$ and homomorphism $\gamma: B \to C$ such that $\gamma(A) = 0$, there is a unique linear map $\delta: B/A \to C$ such that the diagram commutes:
        \begin{center}
            \begin{tikzcd}
                B \arrow[r, "\phi_A"] \arrow[rd, "\gamma"'] & B/A \arrow[d, "\exists ! \delta", dashed] \\
                & C
            \end{tikzcd}
        \end{center}
        Specifically, $\delta$ is defined by mapping $b + A \mapsto \gamma(b)$.
    \end{enumerate}
\end{lem}

\subsection{Construction of the Tensor Product}

\begin{defn}[Constructing $M \otimes N$ and $\otimes$]
    Let $X = M \times N$ be the Cartesian product. Construct the free $R$-module on $X$, denoted $Fr(X)$, whose elements are finite sums of the form:
    \[ \sum \lambda_i (m_i, n_i) \quad \text{where } \lambda_i \in R. \]
    Consider the map $\sigma: M \times N \to Fr(X)$ defined by $(m,n) \mapsto 1(m,n)$. 
    
    For $\sigma$ to be bilinear, we would need equations like $(m_1+m_2, n) - (m_1, n) - (m_2, n) = 0$ and $(\lambda m, n) - \lambda(m, n) = 0$ to hold. However, in $Fr(X)$, these are \emph{not} true! 
    
    To fix this, let $B$ be the submodule of $Fr(X)$ generated by all elements of the form that ``need to be $0$'' to satisfy bilinearity. Define the tensor product as the quotient module:
    \[ M \otimes N = Fr(X)/B \]
    and define the map $\otimes: M \times N \to M \otimes N$ by:
    \[ \otimes: (m,n) \mapsto 1(m,n) + B = m \otimes n. \]
\end{defn}

\begin{claim}
    The map $\otimes$ is bilinear.
\end{claim}
\begin{proof}
    We check one of the bilinearity conditions (the others follow similarly).
    \begin{align*}
        \otimes(\lambda m, n) - \lambda \otimes(m,n) &= 1(\lambda m, n) + B - \lambda(m,n) + B \\
        &= [1(\lambda m, n) - \lambda(m,n)] + B \\
        &= 0 + B
    \end{align*}
    This evaluates to $0 + B$ because the term in the brackets is one of the generators of the submodule $B$, meaning it lives inside $B$. The same logic applies for the other three bilinearity conditions.
\end{proof}

\begin{claim}[Fulfilling the Goal]
    The constructed $M \otimes N$ satisfies the universal property. Let $\psi: M \times N \to Q$ be bilinear. There exists a unique linear $\beta: M \otimes N \to Q$.
\end{claim}
\begin{proof}
    Let $\phi: Fr(X) \to Q$ be the $R$-module homomorphism defined on the basis by the unique $R$-linear extension:
    \[ \phi\left(\sum \lambda_i(m_i, n_i)\right) = \sum \lambda_i \psi(m_i, n_i) \implies \phi(1(m,n)) = \psi(m,n). \]
    We claim $\phi(b) = 0$ for all $b \in B$. Since $B$ is generated by the bilinearity relations, we evaluate $\phi$ on those generators. For instance:
    \[ \phi((\lambda m, n) - \lambda(m,n)) = \psi(\lambda m, n) - \lambda\psi(m,n) = 0 \]
    because $\psi$ is explicitly given as bilinear. Since $\phi$ maps all 4 types of generators of $B$ to $0$, $\phi(B) = 0$.\\
    By our ``Easy Fact", there exists a unique linear map $\beta: \frac{Fr(X)}{B} \to Q$ such that the diagram commutes:
    \begin{center}
        \begin{tikzcd}
            M \times N \arrow[r, "\sigma"] \arrow[rd, "\otimes"'] \arrow[rdd, "\psi"', bend right] & Fr(X) \arrow[d, "\text{quotient}"] \arrow[dd, "\phi", bend left] \\
            & M \otimes N \arrow[d, "\beta", dashed] \\
            & Q
        \end{tikzcd}
    \end{center}
    For elements in $M \otimes N$, we have $m \otimes n = (m,n) + B \implies \beta(m \otimes n) = \psi(m,n)$.
\end{proof}

\begin{rmk}[Uniqueness of $M \otimes N$]
    Suppose $\otimes': M \times N \to M \otimes' N$ is another bilinear map satisfying the universal property. Then, any bilinear map is of the form $\delta \circ \otimes'$ for a linear map $\delta$ on $M \otimes' N$. By universal properties, there is a unique isomorphism $I: M \otimes N \to M \otimes' N$ such that $I(m \otimes n) = m \otimes' n$ for all $m,n$.
\end{rmk}

\subsection{Tensor Product with the Base Ring}

\begin{thm}
    Let $M$ be an $R$-module. Then $R \otimes_R M \cong M$.
\end{thm}
\begin{proof}
    Consider the scalar multiplication map $t: R \times M \to M$ defined by $(r,m) \mapsto rm$. 
    For any bilinear $\psi: R \times M \to Q$, we want to show there exists a unique linear map $\beta: M \to Q$ such that $\psi = \beta \circ t$. 
    
    \begin{center}
        \begin{tikzcd}
            R \times M \arrow[r, "t"] \arrow[rd, "\psi"'] & M \arrow[d, "\exists ! \beta", dashed] \\
            & Q
        \end{tikzcd}
    \end{center}
    If $\beta$ exists, we must have:
    \[ \beta(m) = \beta(1 \cdot m) = \beta(t(1,m)) = \psi(1,m). \]
    This defines $\beta$ uniquely. We verify two claims:
    \begin{enumerate}
        \item \textbf{Claim 1:} $\beta$ is linear because $\psi$ is bilinear.
        \item \textbf{Claim 2:} $\beta \circ t = \psi$. 
        \\ \textit{Proof:} $\psi(r,m) = r\psi(1,m) = \psi(1,rm) = \beta(rm) = \beta(t(r,m))$.
    \end{enumerate}
    By the uniqueness property of the tensor product, we conclude:
    \begin{enumerate}
        \item $R \otimes M \cong M$.
        \item There exists a unique isomorphism $I: R \otimes M \to M$ mapping $r \otimes m \mapsto rm$.
    \end{enumerate}
\end{proof}

        \item There exists a basis $z_1, \dots, z_n$ of $N$ and non-zero elements $d_1, \dots, d_m \in R$ such that $d_i \mid d_{i+1}$ for all $1 \leq i < m$, and $d_1z_1, \dots, d_m z_m$ forms a basis for $M$.

\begin{proof}
    \textbf{Step 1: Construction of direct summands.}\\
    We aim to find an element $z \in N$ ($z \neq 0$) and $d \neq 0$ in $R$ such that $dz \in M$, $Rz$ is a direct summand of $N$, and $R(dz)$ is a direct summand of $M$. 

    First, we claim that if $M \neq 0$, there exists $\psi \in \Hom_R(N, R)$ such that $\psi(M) \neq 0$. To show this, without loss of generality, assume $N = R^n$. Let $m = (a_1, \dots, a_n) \in M$ be a non-zero element. Then $a_j \neq 0$ for some $j$. Using the dual basis functional, $e_j^*(m) = a_j \neq 0$, which implies $e_j^*(M) \neq 0$.

    For any $\phi \in \Hom_R(N, R)$, the image $\phi(M)$ is an ideal of the PID $R$. Consider the set of ideals:
    \[
    \Sigma = \{\phi(M) : \phi \in \Hom_R(N, R)\}.
    \]
    Since $M \neq 0$, $\Sigma$ is a non-empty set containing at least one non-zero ideal. Because $R$ is a PID, it is Noetherian, so $\Sigma$ possesses a maximal element with respect to inclusion. Choose $\psi \in \Hom_R(N, R)$ such that $\psi(M) = (d)$ is maximal in $\Sigma$. Choose $y \in M$ such that $\psi(y) = d$.

    We claim that $d \mid \phi(y)$ for all $\phi \in \Hom_R(N, R)$. To show this, let $d' = \phi(y)$ and let $e = \gcd(d, d')$. Since $R$ is a PID, we can write $e = \lambda d + \mu d'$ for some $\lambda, \mu \in R$. Then:
    \[
    e = \lambda \psi(y) + \mu \phi(y) = (\lambda \psi + \mu \phi)(y) \in (\lambda \psi + \mu \phi)(M).
    \]
    This implies that:
    \[
    (d) \subseteq (e) \subseteq (\lambda \psi + \mu \phi)(M).
    \]
    By the maximality of $(d) = \psi(M)$ in $\Sigma$, we must have $(d) = (e)$. Thus, $d$ and $e$ are associates, which means $d \mid d'$, and consequently $d \mid \phi(y)$.

    Now, let $w_1, \dots, w_n$ be a basis of $N$, and let $w_j^* \in \Hom_R(N, R)$ be the corresponding dual basis functionals. We can express $y = \sum_{j=1}^n \mu_j w_j$, meaning $w_j^*(y) = \mu_j$. By our previous claim, $d \mid w_j^*(y)$, so $d \mid \mu_j$ for all $j$. Let:
    \[
    z = \sum_{j=1}^n \left( \frac{\mu_j}{d} \right) w_j \in N.
    \]
    Notice that $dz = y$. Furthermore, since $\psi(y) = d$, we have $\psi(dz) = d$. By linearity, $d\psi(z) = d$. Since $R$ is an integral domain and $d \neq 0$, we conclude that $\psi(z) = 1$.

    We now claim:
    \begin{enumerate}[label=(\alph*)]
        \item $N = \ker(\psi) \oplus Rz$
        \item $M = (\ker(\psi) \cap M) \oplus Ry$
    \end{enumerate}
    To show (a), note that if $x \in \ker(\psi) \cap Rz$, then $x = rz$ for some $r \in R$, and $0 = \psi(x) = r\psi(z) = r(1) = r$, so $x = 0$. For any $n \in N$, let $\lambda = \psi(n)$. We can write $n = (n - \lambda z) + \lambda z$. Observe that $\psi(n - \lambda z) = \lambda - \lambda \psi(z) = 0$, so $(n - \lambda z) \in \ker(\psi)$ and $\lambda z \in Rz$. Thus $N = \ker(\psi) \oplus Rz$. The proof of (b) is identical, substituting $M$ for $N$ and $y$ for $z$.

    \textbf{Step 2: Proof of Part (1).}\\
    We proceed by induction on the rank of $M$ (with $N$ fixed). If the rank of $M$ is $0$, $M = 0$, which is trivially free. Assume the result holds for submodules of smaller rank. Using our construction, $M = Ry \oplus (\ker(\psi) \cap M)$. Because $Ry$ is generated by a single non-torsion element, it has rank $1$. Thus, the rank of $\ker(\psi) \cap M$ is strictly less than the rank of $M$. By the induction hypothesis, $\ker(\psi) \cap M$ is a free $R$-module. Adding $y$ to a basis of $\ker(\psi) \cap M$ yields a free basis for $M$. Thus $M$ is free of rank $m \leq n$.

    \textbf{Step 3: Proof of Part (2).}\\
    We proceed by induction on the rank $n$ of $N$, proving the statement for all submodules $M \leq N$ simultaneously. If $n=0$, it is trivial. Suppose it holds for free modules of rank $n-1$. From Step 1, $N = \ker(\psi) \oplus Rz$. Since $N$ has rank $n$ and $Rz$ has rank $1$, $\ker(\psi)$ is a free $R$-module of rank $n-1$. By the induction hypothesis applied to the submodule $\ker(\psi) \cap M \leq \ker(\psi)$, there exists a basis $z_2, \dots, z_n$ of $\ker(\psi)$ and elements $d_2, \dots, d_m \in R$ with $d_2 \mid d_3 \mid \cdots \mid d_m$ such that $d_2 z_2, \dots, d_m z_m$ is a basis for $\ker(\psi) \cap M$.

    Let $z_1 = z$ and $d_1 = d$. Then $z_1, z_2, \dots, z_n$ is a basis for $N$, and $d_1 z_1, d_2 z_2, \dots, d_m z_m$ is a basis for $M$. 
    
    It remains only to show that $d_1 \mid d_2$. Define a homomorphism $f: N \to R$ by mapping $z_1 \mapsto 1$, $z_2 \mapsto 1$, and $z_i \mapsto 0$ for $i > 2$. We evaluate $f(M)$: $f(d_1 z_1) = d_1$ and $f(d_2 z_2) = d_2$, so the ideal $f(M)$ contains both $d_1$ and $d_2$, meaning $f(M) = (d_1, d_2)$. Because $(d_1) = \psi(M)$ is maximal among such ideals, and $(d_1) \subseteq (d_1, d_2) = f(M)$, we must have $(d_1) = (d_1, d_2)$. This implies $d_2 \in (d_1)$, so $d_1 \mid d_2$. This completes the proof.
\end{proof}

\begin{defn}
    Let $0 \to A \xrightarrow{\alpha} B \xrightarrow{\beta} C \to 0$ be a short exact sequence of $R$-modules. This sequence \textbf{splits} if and only if there is a homomorphism $\gamma: C \to B$ such that $\beta \circ \gamma = 1_C$. 
\end{defn}

\begin{eg}
    Consider the trivial exact sequence $0 \to A \xrightarrow{\alpha} A \oplus C \xrightarrow{\beta} C \to 0$ with $\alpha: a \mapsto (a,0)$ and $\beta: (a,c) \mapsto c$. Then the map $\gamma: C \to A \oplus C$ given by $\gamma: c \mapsto (0,c)$ splits the sequence. Up to isomorphism, this is the only way a sequence splits. 
\end{eg}

\begin{lem}
    More precisely, if $\gamma: C \to B$ splits the sequence $0 \to A \xrightarrow{\alpha} B \xrightarrow{\beta} C \to 0$, then $\operatorname{im}(\alpha)$ and $\operatorname{im}(\gamma)$ are complementary submodules in $B$, with $\operatorname{im}(\alpha) \simeq A$ and $\operatorname{im}(\gamma) \simeq C$. Thus, $B \simeq A \oplus C$.
    
    \begin{proof}
        By exactness, $\operatorname{im}(\alpha) = \ker(\beta)$. Because $\beta \circ \gamma = 1_C$, we have $\operatorname{im}(\alpha) \cap \operatorname{im}(\gamma) = \ker(\beta) \cap \operatorname{im}(\gamma) = 0$. 
        
        Given any $b \in B$, we can write it as $b = (b - \gamma\beta b) + \underbrace{\gamma\beta b}_{\in \operatorname{im}(\gamma)}$. The first term is in $\operatorname{im}(\alpha)$ because applying $\beta$ gives:
        $$
        \beta(b - \gamma\beta b) = \beta b - \beta\gamma\beta b = \beta b - 1_C(\beta b) = 0.
        $$
        Therefore, $B = \operatorname{im}(\alpha) \oplus \operatorname{im}(\gamma) \simeq A \oplus C$.
    \end{proof}
\end{lem}

\begin{defn}
    An $R$-module $M$ is \textbf{projective} if and only if for every surjective homomorphism $\pi: B \twoheadrightarrow C$ and every homomorphism $\alpha: M \to C$, there exists a homomorphism $g: M \to B$ such that $\pi \circ g = \alpha$. Equivalently, the following diagram commutes:
    $$
    \begin{tikzcd}
        & M \arrow[dl, "g"', dashed] \arrow[d, "\alpha"] \\
        B \arrow[r, "\pi", two heads] & C
    \end{tikzcd}
    $$
\end{defn}

\begin{lem}
    The following are equivalent (TFAE) for an $R$-module $M$:
    \begin{enumerate}
        \item $M$ is projective.
        \item The functor $\operatorname{Hom}_R(M, -)$ is exact. (i.e., for any short exact sequence $0 \to A \to B \to C \to 0$, the induced sequence $0 \to \operatorname{Hom}(M,A) \to \operatorname{Hom}(M,B) \to \operatorname{Hom}(M,C) \to 0$ is exact).
        \item Every short exact sequence $0 \to K \to N \to M \to 0$ ending in $M$ splits.
        \item There exists an $R$-module $N$ such that $M \oplus N$ is a free module.
    \end{enumerate}
\end{lem}

\begin{proof}
    \textbf{$(1) \implies (4)$}: Let $X$ be the underlying set of $M$ and let $F = \operatorname{Free}_R(X)$ with the natural surjection $\pi: F \twoheadrightarrow M$. Applying $(1)$ to $\pi$ and $1_M: M \to M$, there exists $i: M \to F$ such that $\pi \circ i = 1_M$:
    $$
    \begin{tikzcd}
        M \arrow[rr, "1_M"] \arrow[dr, "i"'] & & M \\
        & F \arrow[ur, "\pi"', two heads] &
    \end{tikzcd}
    $$
    Let $N = \ker(\pi)$. By the splitting lemma (Lemma 1), $i(M)$ and $N$ are complementary in $F$. Since $i(M) \simeq M$, we get $F \simeq i(M) \oplus N \simeq M \oplus N$.\\
    \textbf{$(4) \implies (1)$}: Suppose $M \oplus N$ is free. We want to show $M$ is projective. Suppose we are given $B \xrightarrow{\pi} C \to 0$ and $\alpha: M \to C$.
    $$
    \begin{tikzcd}
        & M \oplus N \arrow[dl, "\beta"', dashed] \arrow[d, "\overline{\alpha}"] \\
        B \arrow[r, "\pi", two heads] & C
    \end{tikzcd}
    $$
    Define $\overline{\alpha}: M \oplus N \to C$ by $\overline{\alpha}(m, n) = \alpha(m)$. Let $X$ be a basis of the free module $M \oplus N$. For each $x \in X$, the map $\pi$ is surjective, so we can choose $f(x) \in B$ such that $\pi(f(x)) = \overline{\alpha}(x)$. 
    
    Because $M \oplus N$ is free on $X$, there is a unique homomorphism $\beta: M \oplus N \to B$ extending $f$, meaning $\beta(x) = f(x)$ for all $x \in X$. By choice of $f$, $\pi \beta(x) = \overline{\alpha}(x)$ for all basis elements. Since $X$ generates $M \oplus N$, we have $\pi \circ \beta = \overline{\alpha}$. 
    
    Finally, define $g: M \to B$ by $g(m) = \beta(m, 0)$. Then for any $m \in M$:
    $$
    \pi(g(m)) = \pi(\beta(m, 0)) = \overline{\alpha}(m, 0) = \alpha(m).
    $$
    Thus, $g$ is the required lift, proving $M$ is projective.
\end{proof}

\begin{thm}
    Let $l/k$ be a finite field extension (so $[l:k] < \infty$). Let $\Omega$ be an algebraically closed field, and let $\phi: k \to \Omega$ be a field monomorphism (embedding). There exists at least one monomorphism $\psi: l \to \Omega$ such that $\psi|_k = \phi$. The total number of such extensions $\psi$ is exactly equal to the separable degree $[l:k]_s$. Consequently, the number of extensions is always less than or equal to $[l:k]$.
\end{thm}

\begin{proof}
    We will show this by induction on the degree $n = [l:k]$.

    \textbf{Base Case:} If $[l:k] = 1$, then $l = k$. The only extension of $\phi: k \to \Omega$ to $l$ is $\phi$ itself. Thus, there is exactly $1$ extension, and $1 \leq 1$. The base case holds.

    \textbf{Inductive Step:} Assume the claim holds for all finite extensions of degree less than $n$, and let $[l:k] = n > 1$.

    Choose an element $a \in l \setminus k$. We then have the tower of fields $k \subsetneq k(a) \subseteq l$. Because $a \notin k$, we know $[k(a):k] > 1$. By the Tower Law, $[l:k] = [l:k(a)][k(a):k]$, which implies $[l:k(a)] < [l:k]$.

    \textit{Claim: There are at most $[k(a):k]$ extensions of $\phi$ to an embedding $\phi': k(a) \to \Omega$.}

    To show this, let $m(x) \in k[x]$ be the minimal polynomial of $a$ over $k$, so $\deg(m) = [k(a):k]$. Let $\overline{m}(x) \in \Omega[x]$ be the polynomial obtained by applying $\phi$ to the coefficients of $m(x)$.

    For any embedding $\phi'$ extending $\phi$, we have $m(a) = 0$, which implies that $\overline{m}(\phi'(a)) = 0$. Therefore, $\phi'(a)$ must be a root of $\overline{m}(x)$ in $\Omega$. Because $\Omega$ is a field, the polynomial $\overline{m}(x)$ can have at most $\deg(\overline{m}) = \deg(m) = [k(a):k]$ distinct roots in $\Omega$.

    Since an embedding $\phi': k(a) \to \Omega$ that extends $\phi$ is entirely determined by the value of $\phi'(a)$, there are at most $[k(a):k]$ such extensions $\phi'$.

    Now, by our induction hypothesis, because $[l:k(a)] < [l:k]$, each such embedding $\phi': k(a) \to \Omega$ can be extended to an embedding $\psi: l \to \Omega$ in at most $[l:k(a)]$ ways.

    Multiplying the maximum number of choices at each step, the total number of extensions $\psi: l \to \Omega$ of $\phi$ is at most:
    $$[k(a):k] \times [l:k(a)] = [l:k]$$
    This completes the induction step and the proof.
\end{proof}

\begin{proof}

\begin{thm}
    Let $l/k$, $[l:k]<\infty$, TFAE:
    \begin{enumerate}
        \item $|Aut(l/k)|=[l:k]$
        \item $l/k$ normal and separable
        \item $l/k$ is sfe for some separable $f\in k[x]$
        \item $k=Fix(Aut(l/k))$
    \end{enumerate}
\end{thm}
\begin{proof}
    To show $(2)\implies (3)$, let $l=k(a_1,...,a_t)$, $a_i\in l$. Let $f=\prod_{i=1}^t m_{a_i}^k$. $l/k$ separable implies $f$ is product of separable irreducibles and $f$ is separable. Verify $l/k$ sfe for $f$. $l/k$ normal implies each $m_{a_i}$ splits and $f$ splits. So $l=k( \text{roots of $f$ in $l$})$. \\\\
    To show $(3)\implies (1)$, by induction on $[l:k]$. \\
    (Base Case): $[l:k]=1$ implies $l=k$\\
    (Inductive Step): If $l\neq k$, let $a\in l/k$ be a root of $f$, let $m=m_a^k$. As $f(a)=0$, $m|f$ in $k[x]$. As $l/k$ normal, $m$ splits in $l[x]$. As $l/k$ separable, $m$ separable, $m$ splits as $\prod_{i=1}^t(x-a_i)$ with $a_i$'s distinct. Let $a_1=a$, for each $i$, $m=m_{a_i}$. So for each $i$ $(i=1,2,...,t)$ there is unique $\pi_i$ by 
    \[
    \pi_i:k(a)\to k(a_i) \text{ with } \pi_i|_k=1_k, \pi_i(a)=a_i
    .\]  
    Now $l/k(a)$ sfe for $f$, $l/k(a_i)$ sfe for $f$. $\pi_i(f)=f$. By uniqueness get $e_i:l\to l$ IM, $e_i|_{k(a)}=\pi_i$. Then $e_i\in Aut(l/k)$ with $e_i(a)=a_i$. \\\\
    To show $(1)\implies (2)$, AFSOC that $l/k$ not normal. Let $a\in l$ be such that $m_a$ fails to split. Then number of roots of $m_a<[k(a):k]=\deg(m_a)$. List roots of $m_a$ in $l$ as $a_i$f or $i=1,...,t$, $a_1=a$. If $\rho\in Aut(l/k), m(a)=0\implies m(\rho(a))=0\implies \rho(a)=a_i$ for some $1\leq i \leq t$. \\
    So $\rho|_{k(a)}=\pi_i$ for some $\pi_i$ unique IM $\pi_i:k(a)\simeq k(a_i)$. Additionally, $\pi_i|_k=1_k$. \\
    $\rho$ extends $\pi_i:k(a)\hookrightarrow l$ and the number of such extensions is at most $[l:k(a)]$. So $\#$ of $\rho$ is at most $t[l:k(a)]<[l:k]$ a contradiction. \\
    If $l/k$ normal but not separable, let $a\in l$ be such that $m_a$ not separable, $m_a$ splits in $l[x]$ so $m_a$ has repeated roots in $l$, so the number of roots is less than $[k(a):k]=\deg(m_a)$. \\\\
    To show $(1)\implies (4)$, we'll prove a stronger statement: Let $l$ be field, let $G\leq Aut(l)$, $|G|<\infty, |G|=n$, $k=Fix(G)$ then 
    \begin{enumerate}
        \item $[l:k]=n$
        \item $G=Aut(l/k)$
        \item $l/k$ is a Galois extension.
    \end{enumerate}
    We have $G\leq Aut(l/k)$ so not possible that $[l:k]<n$. \\
    Suppose for contradiction that $[l:k]>n$. Find $a_1,...,a_{n+1}\in l$ linearly indepdnent over $k$. Enumerate $G$ as $\rho_1,...,\rho_n$ with $\rho_1=1_l$. For $j=1,...,n+1$, 
    \[
    v_j= \begin{pmatrix}
        {\rho_1(a_j)} \\ 
        {\vdots} \\ 
        {\rho_n(a_j)}
    \end{pmatrix}\in l^n
    .\] 
    $l^n$ has dimension $n$ as a vector space over $l$. So there is a non-trivial linear dependence among $v_1,...,v_{n+1}$. Let $s$ be minimal size of dependent subset. So relabeling, WLOG $v_1,...,v_s$ are dependent. So 
    \[
    \sum_{i=1}^s\lambda_i v_i=0\implies \exists j \text{ such that } \lambda_j\neq 0
    .\] 
    As $s$ minimal, all $\lambda_i\neq 0$. Scaling by $\lambda_1^{-1}, \lambda_1=1$. \\
    For all $i$,
    \[
    \sum_{j=1}^s \lambda_j \rho_i(a_j)=0 \implies \sum_{j=1}^s \lambda_j a_j=0 \tag{$(*)$}
    .\] 
    Not all $\lambda_j\in k=Fix(G)$ as $a_j$ linearly independent over $k$. Relabeling so that $\lambda_s\notin k, \rho_n(\lambda_s)\neq \lambda_s$.\\
    Applying $\rho_n$ to $(*)$ get 
    \[
    \sum_{j=1}^s \rho_n(\lambda_j)\rho_n\rho_i(a_j)=0 \text{ for all $i$}
    .\]  
    As $G$ group, $(\rho_n\rho_i)_{i=1,...,n}$ spans $G$ so 
    \[
    \sum_{j=1}^s\rho_n(\lambda_j)\rho_i(a_j)\neq 0 \text{ for all $i$}
    .\] 
    so in particular,
    \[
    \sum_{j=1}^s\rho_n(\lambda_j)v_j=0
    .\] 
    Subtracting, 
    \[
    \sum_{j=1}^s(\lambda_j-\rho_n(\lambda_j))v_j=0
    .\] 
    We have $\lambda_1=1$ and $\lambda_s\neq \rho(\lambda_s)$ so it has non-trivial dependence among solution $v$'s. Thus a contradiction. \\
    So we've shown $[l:k]=n$ so as $G\leq Aut(l/k), |G|=n, |Aut(l/k)|\leq n$ so $G=Aut(l/k)$. \\
    As $|Aut(l/k)|=[l:k]$, $l/k$ is galois satisfying $(1)$. \\
    To finish equivalence, let $G=Aut(l/h)$ and let $k'=Fix(G)$. By preceding theorem, $G=Aut(l/h')$ and $[l:k']=|G|$ then we have 
    \[
    (1)\implies |G|=[l:k]\iff k'=k\iff (4)
    .\] 
\end{proof}
\begin{defn}
    For $l/k$, $[l:k]<\infty$, $l/k$ is Galois if it has any of these equivalent properties. 
\end{defn}
\begin{thm}
    Fundamental Theorem of Galois Theory: If $l/k, [l:k]<\infty$, $l/k$ is Galois then operations 
    \[
    k'\mapsto Aut(l/k') \text{ and } H\mapsto Fix(H)
    .\] 
    are mutually inverse.\\
    Note: First operation tells us $k'$ intermediate and second $H\leq Aut(l/k)$. 
\end{thm}
\begin{thm}
    Fundamental Theorem of Galois Theory: Let $l/k$ be Galois then 
    \begin{enumerate}
        \item $l/k'$ is Galois for all intermediate fields.
        \item Operations 
        \[
    k'\mapsto Aut(l/k') \text{ and } H\mapsto Fix(H)
    .\] 
    are mutually inverse.\\
    Note: First operation tells us $k'$ intermediate and second $H\leq Aut(l/k)$. 
    \item For $h'$ intermediate, $k'/k$ is Galois iff $Aut(l/k')\normal Aut(l/h)$ and in this case 
    \[
    Aut(k'/k)\simeq \frac{Aut(l/k)}{Aut(l/k')}
    .\] 
    \end{enumerate}
\end{thm}
\begin{proof}
    To show each one:
    \begin{enumerate}
        \item $l/k$ sfe of $k$ for separate $f\in k[x]$ then $f$ separable in $h'[x]$, $l/k'$ sfe for $f$ so $l/h'$ Galois. 
        \item Let $k'$ be intermediate, then as $l/k'$ Galois, $k'=Fix(Aut(l/k'))$. Conversely, let $H\leq Aut(l/h)$ then $H$ is finite so by preceding theorem $H=Aut(l/Fix(H))$ 
        \item Let $k'$ be intermediate with $k'/k$ Galois using normality shows for all $\rho\in Aut(l/h)$ we have $\rho|_{k'}\in Aut(h'/k)$. We have HM $\rho\mapsto \rho|_{k'}$, $Aut(l/h)\to Aut(k'/k)$ with kernel $Aut(l/h')$. 
    \end{enumerate}
\end{proof}

\section{Lecture 35: Radical}

We complete the proof of the Fundamental Theorem of Galois Theory, Part (3): $L/K$ Galois, $[L:K]<\infty$, $K'$ intermediate. Then $K'/K$ is Galois $\iff$ $\Aut(L/K')\normal \Aut(L/K)$, and in this case 
\[
\Aut(K'/K)\cong \frac{\Aut(L/K)}{\Aut(L/K')}
.\]
\begin{proof}
    \textbf{($\Rightarrow$):} Assume $K'/K$ Galois. Let $\sigma\in\Aut(L/K)$, $a\in K'$ such that $m_a$ splits in $K'[x]$:
    \[
    m_a := \prod_{i=1}^t(x-a_i),\quad a_i\in K'
    .\]
    For all $\sigma\in\Aut(L/K)$, $m(a)=0\implies m(\sigma(a))=0$, so $\sigma(a)=a_i$ for some $i$. Therefore $\sigma|_{K'}\in\Aut(K'/K)$, and $\sigma\mapsto\sigma|_{K'}$ is a homomorphism
    \[
    \Aut(L/K)\to\Aut(K'/K),\quad \text{kernel}=\Aut(L/K')
    .\]
    Since $L/K$ and $L/K'$ are both Galois,
    \[
    |\Aut(L/K)|=[L:K],\quad |\Aut(L/K')|=[L:K']
    .\]
    So 
    \[
    \frac{|\Aut(L/K)|}{|\Aut(L/K')|}=[K':K]
    .\]
    Thus $|\im(\text{restriction})|=[K':K]$ and $\im=\Aut(K'/K)$. By the first isomorphism theorem,
    \[
    \Aut(K'/K)\cong \frac{\Aut(L/K)}{\Aut(L/K')}
    .\]
    In particular $\Aut(L/K')\normal \Aut(L/K)$.

    \textbf{($\Leftarrow$):} We saw $\Aut(L/K')\normal\Aut(L/K)$ if $K'/K$ Galois. Now suppose $\Aut(L/K')\normal\Aut(L/K)$; we want to show $K'/K$ is Galois.

    Since $L/K$ is separable, $K'/K$ is separable for every intermediate $K'$. It remains to show $K'/K$ is normal.

    Let $a\in K'$, $m=m_a^K$ the minimal polynomial. Since $L/K$ is normal, $m$ splits in $L[x]$:
    \[
    m=\prod(x-a_i)
    .\]
    By HWII, for all $i$ there exists $\sigma\in\Aut(L/K)$ such that $\sigma(a)=a_i$.
    \begin{claim}
        $a_i\in K'$ for all $i$.
    \end{claim}
    \begin{proof}
        Since $L/K'$ is Galois, $K'=Fix(\Aut(L/K'))$. So it suffices to show $a_i\in Fix(\Aut(L/K'))$.

        Fix $\sigma\in\Aut(L/K)$ with $\sigma(a)=a_i$. Fix $\tau\in\Aut(L/K')$. Since $\Aut(L/K')\normal\Aut(L/K)$, we have $\sigma^{-1}\tau\sigma\in\Aut(L/K')$. Since $a\in K'$:
        \[
        (\sigma^{-1}\tau\sigma)(a)=a
        .\]
        Therefore $\tau\sigma(a)=\sigma(a)$, i.e., $\tau(a_i)=a_i$.
    \end{proof}
    Since all roots $a_i$ of $m_a^K$ lie in $K'$, $m$ splits in $K'[x]$, so $K'/K$ is normal. Combined with separability, $K'/K$ is Galois.
\end{proof}

\begin{defn}[Radical Extension]
    A field extension $L/K$ is \textbf{radical} if and only if there exist intermediate fields
    \[
    K=K_0\subseteq K_1\subseteq K_2\subseteq \cdots \subseteq K_s=L
    \]
    such that for all $i$, $K_{i+1}=K_i(\alpha_i)$ where there exists $n_i>0$ with $\alpha_i^{n_i}\in K_i$.
\end{defn}

\begin{defn}[Solvable by Radicals]
    A polynomial $f\in K[x]$ is \textbf{solvable by radicals} if and only if there exists a radical extension $L/K$ with an intermediate field that is a splitting field extension of $K$ for $f$.
\end{defn}

Until further notice, all fields have characteristic $0$.

\begin{thm}
    The following are equivalent:
    \begin{enumerate}
        \item $f$ is solvable by radicals.
        \item If $L/K$ is the splitting field extension over $K$ for $f$, then $\Aut(L/K)$ is solvable.
    \end{enumerate}
\end{thm}

\begin{thm}
    For any field $K$ (any characteristic), any finite subgroup of $K^*$ (under multiplication) is cyclic.
\end{thm}

\begin{defn}[Primitive Root of Unity]
    For any field $K$ (any characteristic), a \textbf{primitive $n$-th root of unity} is an element $a\in K^*$ that is a generator of $\{a : a^n=1\}$, i.e., $a$ has order $n$ in $K^*$. Equivalently, $\mathrm{ord}(a)=n$ in $K^*$.
\end{defn}

\begin{eg}
    In $\C$, $e^{2\pi i/n}$ is a primitive $n$-th root of unity.
\end{eg}

Let $K$ have characteristic zero. Consider $f(x):=x^n-1\in K[x]$. Then $f'(x)=nx^{n-1}\neq 0$ since $\mathrm{char}(K)=0$, so $f$ has no repeated roots in any extension $L/K$, hence $f$ is separable.

Let $L/K$ be the splitting field extension for $x^n-1$. Since $f$ splits without repeated factors, there are exactly $n$ distinct $n$-th roots of $1$ in $L$. The set $\{a\in L^*: a^n=1\}\leq L^*$ is a finite subgroup, hence cyclic of order $n$, generated by some primitive $n$-th root of unity $\alpha$. Moreover $L=K(\alpha)$.

Let $\sigma\in\Aut(L/K)$. Then $\sigma$ is determined by $\sigma(\alpha)$, which must also be a primitive $n$-th root, so $\sigma(\alpha)=\alpha^i$ for some $i$. It follows that $\Aut(L/K)$ is abelian: for any $\sigma,\tau\in\Aut(L/K)$, $\sigma\circ\tau=\tau\circ\sigma$ since both map $\alpha$ to the same power.

More generally, consider $g(x):=x^n-a$ where $a\in K$, $a\neq 0$. We have $g'(x)=nx^{n-1}\neq 0$, so $g$ has no repeated roots. Let $L/K$ be the SFE for $g$. In $L[x]$,
\[
x^n-a=\prod_{i=1}^n(x-\alpha_i),\quad \alpha_i^n=a,\quad \alpha_i\text{ distinct}
.\]
Consider $\frac{\alpha_i}{\alpha_1}$: we have $\left(\frac{\alpha_i}{\alpha_1}\right)^n=\frac{a}{a}=1$. So $x^n-1$ splits in $L$, giving $n$ distinct $n$-th roots of unity. If $\omega$ is a primitive $n$-th root of unity and $\alpha^n=a$, then
\[
x^n-a=\prod_{i=0}^{n-1}(x-\omega^i\alpha)
.\]
So $L=K(\omega^i\alpha\mid i<n)=K(\omega,\alpha)$.

In all cases, $K\subseteq K(\omega)\subseteq K(\omega,\alpha)$, where $\Aut(K(\omega)/K)$ is abelian and $\Aut(K(\omega,\alpha)/K(\omega))$ is abelian. Therefore $\Aut(L/K)$ is solvable.
